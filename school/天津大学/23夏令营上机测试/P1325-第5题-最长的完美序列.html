# 题目内容

小C学习数组时非常喜欢取子数组这一操作，即选择一个起始点一个终止点，将其中连续地取出形成子数组。现小C想对子数组进行约束，他想知道如果子数组中所有元素的最大值和最小值的差不超过k，他所选子数组的最长长度是多少。

# 输入

多组样例，第一行包含一个整数TTT表示样例个数。 每组样例第一行包含两个整数nnn kkk，分别表示原数组的长度和对子数组的约束。第二行包含nnn个由空格分离的整数，表示数组元素。

# 输出

对于每一组输入，输出一个整数表示小CCC可以取到最长子数组的长度。

# 样例

输入

2
9 10
19 18 16 18 6 6 11 13 14
5 8
17 19 13 18 14

输出

5
5

提示

第一组样例中可以取666 666 111111 131313 141414作为子数组。

第二组样例中可以取171717 191919 131313 181818 141414作为子数组。

T≤50T\leq 50T≤50

数据保证数组长度不超过500005000050000，数组元素在intintint范围内。

# 思路：双指针+(multiset/st表)

一道经典的区间限制题。

枚举区间的右端点，然后不断移动左端点直到满足要求。

这里的要求是区间的最大值减最小值小于等于 kkk 。 则需要一个可以快速求解区间极值的数据结构：

  • 线段树可以 O(logn)O(\log n)O(logn) 求解。
  • st表可以通过 O(nlogn)O(n\log n)O(nlogn) 的预处理，然后 O(1)O(1)O(1) 求出区间极值
  • 使用默认 comparator，multiset.begin() 是 set 中最小值，multiset.rbegin() 是 set 中最大值。

故可以考虑上述三种方式，线段树常数较大，下面提供 st表 和 multiset 的实现。

时间复杂度：O(nlogn)O(n\log n)O(nlogn)

# 代码

//multiset 维护
#include<bits/stdc++.h>
using namespace std;

void solve() {
    int n, k;
    cin >> n >> k;
    multiset<int> S;

    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    int ans = 0;
    for (int r = 0, l = 0; r < n; ++r) {
        // 添加区间右端点元素
        S.insert(a[r]);
        // 不断判断区间内最大值减最小值是否小于等于k
        while (l <= r && *S.rbegin() - *S.begin() > k) {
            // 不满足则右移左端点
            S.erase(S.find(a[l]));
            l += 1;
        }
        // 最后一定是满足区间最大值减最小值小于等于k的，更新答案
        ans = max(ans, r - l + 1);
    }

    cout << ans << "\n";
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T = 1;
    cin >> T;
    while (T--) {
        solve();
    }

    return 0;
}

// st表维护
#include<bits/stdc++.h>
using namespace std;

const int N = 50010;
const int BIT = 16;
// st_max[i][j] 表示从 [i, i+(1<<j)-1] 这长度为 (1<<j) 的区间的最大值
int st_max[N][BIT];
// st_min[i][j] 表示从 [i, i+(1<<j)-1] 这长度为 (1<<j) 的区间的最小值
int st_min[N][BIT];

void solve() {
    int n, k;
    cin >> n >> k;

    for (int i = 0; i < n; ++i) {
        cin >> st_max[i][0];
        st_min[i][0] = st_max[i][0];
    }

    for (int j = 1; j < BIT; ++j)
        for (int i = 0; i + (1 << j) - 1 < n; ++i) {
            // 倍增求区间极值
            st_max[i][j] = max(st_max[i][j - 1], st_max[i + (1 << (j - 1))][j - 1]);
            st_min[i][j] = min(st_min[i][j - 1], st_min[i + (1 << (j - 1))][j - 1]);
        }

    // lg[i] 表示以 2 为底的 log(i) 的下取整的值
    vector<int> lg(n + 1);
    for (int i = 2; i <= n; ++i) lg[i] = lg[i / 2] + 1;

    int ans = 0;
    for (int r = 0, l = 0; r < n; ++r) {
        while (l <= r) {
            int bit = lg[r - l + 1];
            // [l, r] 的最大值一定是 [l, l+(1<<bit)-1] 和 [l+(1<<bit), r] 中的较大值
            int maxv = max(st_max[l][bit], st_max[r - (1 << bit) + 1][bit]);
            int minv = min(st_min[l][bit], st_min[r - (1 << bit) + 1][bit]);
            if (maxv - minv > k) {
                l += 1;
            } else {
                break;
            }
        }
        ans = max(ans, r - l + 1);
    }

    cout << ans << "\n";
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T = 1;
    cin >> T;

    for (int ca = 1; ca <= T; ++ca) {
        solve();
    }

    return 0;
}
