# 题面

给出一个长度为nnn的数列和一一个长度为mmm的数列，可以构造得到一个n×mn\times mn×m的矩阵C,其中Ci,j=Ai×BjC_{i,j}= A_i \times B_jC​i,j​​=A​i​​×B​j​​。

给出整数KKK，你需要求出CCC中的第kkk大的数的值。

# 输入格式

第一行输入三个整数n,m,Kn, m,Kn,m,K(1≤n,m≤105,1≤K≤n∗m1 \leq n,m \leq 10^5,1\leq K \leq n * m1≤n,m≤10​5​​,1≤K≤n∗m)。

第二行输入nnn个空格隔开的整数A1,A2,...,An(−106≤Ai≤106)A_1,A_2,...,A_n(-10^6\leq A_i\leq 10^6)A​1​​,A​2​​,...,A​n​​(−10​6​​≤A​i​​≤10​6​​)

第三行输入mmm个空格隔开的整数B1,B2,...,Bm(−106≤Bi≤106)B_1, B_2,...,B_m(-10^6\leq B_i\leq 10^6)B​1​​,B​2​​,...,B​m​​(−10​6​​≤B​i​​≤10​6​​)

# 输出格式

输出一行一 个整数，表示矩阵中的第KKK大的数的值。

# 样例

input

3 3 7
5 4 1
4 5 3

output

5

# 解法一：

二分答案为 valvalval ，计算小于等于 valvalval 的乘积个数。

如此我们需要考虑所有乘积的情况，因为是求第 kkk 大，需要优先考虑正数。但是第 kkk 大等价于第 n×m−k+1n\times m - k + 1n×m−k+1 小。所以可以将问题转换为求第 kkk 小，这里优先考虑负数。本质上来说两者都是一样的。

考虑负数部分，Apos×BnegA_{pos}\times B_{neg}A​pos​​×B​neg​​ 和 Aneg×BposA_{neg}\times B_{pos}A​neg​​×B​pos​​ 两组。

# 解法二：

考虑是否可以继续优化，是否可以将 checkcheckcheck 函数内的二分给优化掉。

在解法一中，我们需要二分的原因是，我们枚举每个 AAA 中的值 xxx，然后根据 xxx 的正负性去二分有序的 BBB 。这里我们并没有考虑到 AAA 的单调性。

以 Apos×BnegA_{pos}\times B_{neg}A​pos​​×B​neg​​ 为例，在解法一中我们已经讨论过，因为 AposA_{pos}A​pos​​ 乘 BnegB_{neg}B​neg​​ 是正数乘负数，我们枚举 AposA_{pos}A​pos​​ 的每个数 xxx ，然后对 BposB_{pos}B​pos​​ 进行二分，因为正数乘负数时，负数越小，乘积越小，所以我们让 BnegB_{neg}B​neg​​ 单调递增。对于正数乘负数，当负数固定时，正数越大，则乘积越小，故我们让 AposA_{pos}A​pos​​ 单调递减。

这有什么用呢？

因为我们是按顺序枚举的，从前往后的 AposA_{pos}A​pos​​ 是单调递减的，而 BnegB_{neg}B​neg​​ 是单调递增的。我们令 AposA_{pos}A​pos​​ 中的数为 xxx ，BnegB_{neg}B​neg​​ 中的数为 yyy ，当 xxx 增大时，yyy 不能再增大，yyy 只能不变或减小，这样保证 BnegB_{neg}B​neg​​ 中小于等于 yyy 的数和 xxx 的乘积还是小于等于 valvalval 的。

如此，就是一个典型的双指针了。

从前往后枚举 xxx ，因为 BnegB_{neg}B​neg​​ 单调递增，从后往前枚举 yyy，直到使得 x×y≤valx\times y\leq valx×y≤val，因为 yyy 之前的所有数都不比 yyy 大，与 xxx 的乘积必然 ≤val\leq val≤val，这样就可以计算出 AposA_{pos}A​pos​​ 中的每个数可以和 BnegB_{neg}B​neg​​ 中多少个数的乘积 ≤val\leq val≤val 了。

如此，我们将 checkcheckcheck 部分的二分优化掉，checkcheckcheck 函数的复杂度为 O(n)O(n)O(n) ，总时间复杂度为 O(nlogn)O(n\log n)O(nlogn)。

# 代码

// O(nlogn)
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    ll n, m, k;
    cin >> n >> m >> k;
    // Apos 是 A 数组中的正数，Aneg 是 A 数组中的负数
    // Bpos 是 B 数组中的正数，Bneg 是 B 数组中的负数
    vector<ll> Apos, Aneg, Bpos, Bneg;
    ll A0 = 0, B0 = 0;

    for (int i = 0; i < n; ++i) {
        int x; cin >> x;
        if (x > 0) Apos.emplace_back(x);
        else if (x < 0) Aneg.emplace_back(x);
        else A0 += 1;
    }

    for (int i = 0; i < m; ++i) {
        int x; cin >> x;
        if (x > 0) Bpos.emplace_back(x);
        else if (x < 0) Bneg.emplace_back(x);
        else B0 += 1;
    }

    // 先排序，后续如果要使得某个单调递增或者单调递减，只需要 O(n) 的复杂度 reverse 即可
    sort(Apos.begin(), Apos.end());
    sort(Aneg.begin(), Aneg.end());
    sort(Bpos.begin(), Bpos.end());
    sort(Bneg.begin(), Bneg.end());

    // 乘积为 0 的个数
    long long zero = A0 * m + B0 * n - A0 * B0;

    // 双指针，注意这里的 k 传的为引用，修改原值
    auto check2 = [&](vector<ll>& a, vector<ll>& b, ll& k, ll val) {
        if (a.empty() || b.empty()) return false;

        // a[0] * b[0] 一定是最小的
        for (int i = 0, j = int(b.size()) - 1; i < a.size() && j >= 0; ++i) {
            while (j >= 0 && a[i] * b[j] > val) j -= 1;
            k -= (j + 1);
        }
        return k <= 0;
    };

    auto check = [&](ll val, ll k) {
        // 负数，Apos * Bneg
        if (!Apos.empty() && !Bneg.empty()) {
            // Apos 是递减的，Bneg 是递增的
            if (Apos.front() < Apos.back()) reverse(Apos.begin(), Apos.end());
            if (Bneg.front() > Bneg.back()) reverse(Bneg.begin(), Bneg.end());
            if (check2(Apos, Bneg, k, val)) return true;
        }

        // 负数，Aneg * Bpos
        if (!Bpos.empty() && !Aneg.empty()) {
            // Aneg 是递增，Bpos 是递减的
            if (Aneg.front() > Aneg.back()) reverse(Aneg.begin(), Aneg.end());
            if (Bpos.front() < Bpos.back()) reverse(Bpos.begin(), Bpos.end());
            if (check2(Aneg, Bpos, k, val)) return true;
        }

        // 如果 val < 0 ，则不必进行下面乘积大于等于 0 的计算了
        if (val < 0) return k <= 0;
        
        // 零
        k -= zero;
        if (k <= 0) return true;

        // 正数，Apos * Bpos
        if (!Apos.empty() && !Bpos.empty()) {
            // Apos 是递增的，Bpos 是递增的
            if (Apos.front() > Apos.back()) reverse(Apos.begin(), Apos.end());
            if (Bpos.front() > Bpos.back()) reverse(Bpos.begin(), Bpos.end());
            if (check2(Apos, Bpos, k, val)) return true;
        }

        // 正数，Aneg * Bneg
        if (!Aneg.empty() && !Bneg.empty()) {
            // Aneg 是递减的，Bneg 是递减的
            if (Aneg.front() < Aneg.back()) reverse(Aneg.begin(), Aneg.end());
            if (Bneg.front() < Bneg.back()) reverse(Bneg.begin(), Bneg.end());
            if (check2(Aneg, Bneg, k, val)) return true;
        }

        return k <= 0;
    };

    // 从第 k 大转换为 第 n * m - k + 1 小
    k = n * m - k + 1;
    ll l = -1e12, r = 1e12;
    while (l < r) {
        long long mid = (l + r) >> 1;
        if (check(mid, k)) r = mid;
        else l = mid + 1;
    }


    cout << l << "\n";

    return 0;
}
