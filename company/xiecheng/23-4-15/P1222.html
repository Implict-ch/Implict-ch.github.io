# 题目内容

塔子哥最近在进行一项自然语言处理的研究项目，需要对一些文本数据进行分析。

其中，他得到了一个字符矩阵，该矩阵包含大量的文本数据。在对数据进行初步分析时，他发现在这个字符矩阵中，经常会同时出现 y、o和u这三个字符，而且它们的位置总是相邻的。

为了更加深入地研究这个现象，他决定统计一下在这个字符矩阵中有多少个 2×22\times 22×2 的子矩阵同时包含 y、o和u这三种字符。

# 输入描述

第一行输入两个正整数 nnn 和 mmm ，代表矩阵的行数和列数。

接下来的 nnn 行，每行输入一个长度为 mmm 的、仅由英文小写字母组成的字符串，代表塔子哥拿到的矩阵。

1≤n,m≤1031 \le n, m \le 10^31≤n,m≤10​3​​

# 输出描述

一个整数，代表同时包含了 y 、 o 和 u 三种字符的 2×22\times 22×2 的子矩阵数量。

# 样例

输入

4 4
youy
ouyo
uyoy
oyuo

输出

8

# 思路:模拟

我们可以通过暴力枚举每个 2×2子矩阵来解决问题。对于每个子矩阵，我们只需要判断是否同时存在字符 yyy、ooo 和uuu，如果是，则计数加一。总共需要枚举 (n-1)×(m-1)个子矩阵。

具体步骤如下：

  1. 初始化计数器 ansansans 为000。
  2. 使用两层循环遍历每个子矩阵的起始位置 iii 和 jjj，范围为 000 到 n−2n-2n−2和 000 到 m−2m-2m−2。
  3. 对于当前的子矩阵，构建一个字符串sss，包含该子矩阵的所有字符。
  4. 在字符串 sss 中判断是否同时包含字符 yyy、ooo和 uuu，如果是，则将计数器 ansansans 加一。

# 代码

# CPP

#include <iostream>
#include <string>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    string s1[1004];
    int ans = 0;
    for (int i = 0; i < n; i++) {
        cin >> s1[i];
    }
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < m - 1; j++) {
            string s;
            for (int k = 0; k < 2; k++) {
                for (int k2 = 0; k2 < 2; k2++) {
                    s += s1[i + k][j + k2];
                }
            }
            if (s.find("y") != string::npos && s.find("o") != string::npos && s.find("u") != string::npos) {
                ans += 1;
            }
        }
    }
    cout << ans << endl;
    return 0;
}

# Go

package main

import (
	"fmt"
	"bufio"
	"os"
	"strings"
)

func main() {
	var n, m int
	fmt.Scanf("%d %d\n", &n, &m)
	
	reader := bufio.NewReader(os.Stdin)
	
	s1 := make([]string, n)
	
	for i := 0; i < n; i++ {
		s1[i], _ = reader.ReadString('\n')
		s1[i] = strings.TrimSuffix(s1[i], "\n")
	}
	
	ans := 0
	
	for i := 0; i < n-1; i++ {
		for j := 0; j < m-1; j++ {
			s := ""
			for k := 0; k < 2; k++ {
				for k2 := 0; k2 < 2; k2++ {
					s += string(s1[i+k][j+k2])
				}
			}
			if strings.Contains(s, "y") && strings.Contains(s, "o") && strings.Contains(s, "u") {
				ans++
			}
		}
	}
	fmt.Println(ans)
}
