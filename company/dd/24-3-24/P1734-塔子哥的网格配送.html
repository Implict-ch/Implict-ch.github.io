# 问题描述

在一个 nnn 行 mmm 列的网格城市中，配送员塔子哥需要从起点 AAA 前往取餐点 BBB，再将餐食送至客户处 CCC。网格中的部分格子是不可通行的。塔子哥每次移动可以从一个可通行的格子走到另一个相邻的可通行格子。已知塔子哥的起点坐标为 (XA,YA)(X_A, Y_A)(X​A​​,Y​A​​)，取餐点坐标为 (XB,YB)(X_B, Y_B)(X​B​​,Y​B​​)，客户处坐标为 (XC,YC)(X_C, Y_C)(X​C​​,Y​C​​)。请计算塔子哥完成配送任务至少需要移动多少步。

# 输入格式

第一行包含两个正整数 n,mn, mn,m，表示网格城市的行数和列数。
接下来 nnn 行，每行包含 mmm 个字符，表示网格图。其中 # 表示不可通行的格子，. 表示可通行的格子。
最后一行包含六个正整数 XA,YA,XB,YB,XC,YCX_A, Y_A, X_B, Y_B, X_C, Y_CX​A​​,Y​A​​,X​B​​,Y​B​​,X​C​​,Y​C​​，表示起点、取餐点和客户处的坐标。

# 输出格式

输出一个整数，表示塔子哥完成配送任务的最少移动步数。

# 样例输入

7 8
........
.#.#####
.#...#..
.#.#.#.#
.#.#....
##.###.#
.....#..
5 1 1 7 2 3

# 样例输出

15

# 评测数据与规模

对于所有评测数据，满足 1≤XA,XB,XC≤n1 \leq X_A, X_B, X_C \leq n1≤X​A​​,X​B​​,X​C​​≤n 且 1≤YA,YB,YC≤m1 \leq Y_A, Y_B, Y_C \leq m1≤Y​A​​,Y​B​​,Y​C​​≤m，1≤n,m≤10001 \leq n, m \leq 10001≤n,m≤1000。保证起点、取餐点和客户处的坐标位于可通行的格子上，并且至少存在一条合法的路径连接这三点。

# 题解

​ BFS 的经典题，如果之前写过 BFS 走迷宫的同学，这题应该可以轻松 AC。

走迷宫具体做法如下

​ 从起点开始，往前走第一步，记录下所有第一步能走到的点，然后从所第一步能走到的点开始，往外扩展一层，这是第二步。

记录下所有第二步能走到的点，重复下去，直到走到终点。输出步数即可。

对于本题

​ 对于本题来说只是中间新增了一个点，那么我们只需要分两次走就行。

  1. 从起点走到中间点 BFS 一次
  2. 从中间点走到终点 BFS 一次

最终把两次答案相加即可

    import sys
input = lambda:sys.stdin.readline().strip()
from collections import deque
n, m = map(int, input().split())
dx = [-1, 0, 1, 0]
dy = [0, 1, 0, -1]
g = []
for _ in range(n):
    g.append(input())
xa, ya, xb, yb, xc, yc = map(int, input().split())
def bfs(stx, sty, end_x, end_y) -> int:
    dist = [[-1] * m for _ in range(n)]
    q = deque()
    dist[stx][sty] = 0
    q.append((stx, sty))
    while len(q):
        x, y = q.popleft()
        if x == end_x and y == end_y:
            break
        for i in range(4):
            nx, ny = x + dx[i], y + dy[i]
            if nx < 0 or nx >= n or ny < 0 or ny >= m:
                continue
            if dist[nx][ny] == -1 and g[nx][ny] != '#':
                dist[nx][ny] = dist[x][y] + 1
                q.append((nx, ny))
    return dist[end_x][end_y]
first_ans = bfs(xa - 1, ya - 1, xb - 1, yb - 1)
second_ans = bfs(xb - 1, yb - 1, xc - 1, yc - 1)
print(first_ans + second_ans)

    import java.util.*;
import java.io.*;

public class Main {
    static int n, m;
    static int[] dx = {-1, 0, 1, 0};
    static int[] dy = {0, 1, 0, -1};
    static char[][] g;

    public static int bfs(int stx, int sty, int end_x, int end_y) {
        int[][] dist = new int[n][m];
        for (int[] row : dist) Arrays.fill(row, -1);
        Queue<int[]> q = new LinkedList<>();
        dist[stx][sty] = 0;
        q.offer(new int[]{stx, sty});
        
        while (!q.isEmpty()) {
            int[] curr = q.poll();
            int x = curr[0], y = curr[1];
            if (x == end_x && y == end_y) {
                break;
            }
            for (int i = 0; i < 4; i++) {
                int nx = x + dx[i], ny = y + dy[i];
                if (nx >= 0 && nx < n && ny >= 0 && ny < m && dist[nx][ny] == -1 && g[nx][ny] != '#') {
                    dist[nx][ny] = dist[x][y] + 1;
                    q.offer(new int[]{nx, ny});
                }
            }
        }
        return dist[end_x][end_y];
    }

    public static void main(String[] args) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        String[] firstLine = reader.readLine().split(" ");
        n = Integer.parseInt(firstLine[0]);
        m = Integer.parseInt(firstLine[1]);
        g = new char[n][m];

        for (int i = 0; i < n; i++) {
            g[i] = reader.readLine().toCharArray();
        }
        
        String[] coords = reader.readLine().split(" ");
        int xa = Integer.parseInt(coords[0]) - 1;
        int ya = Integer.parseInt(coords[1]) - 1;
        int xb = Integer.parseInt(coords[2]) - 1;
        int yb = Integer.parseInt(coords[3]) - 1;
        int xc = Integer.parseInt(coords[4]) - 1;
        int yc = Integer.parseInt(coords[5]) - 1;

        int first_ans = bfs(xa, ya, xb, yb);
        int second_ans = bfs(xb, yb, xc, yc);
        
        System.out.println(first_ans + second_ans);
    }
}

    #include <iostream>
#include <vector>
#include <queue>
#include <cstring>
using namespace std;

const int MAXN = 1000;
int n, m;
int dx[] = {-1, 0, 1, 0};
int dy[] = {0, 1, 0, -1};
string g[MAXN];

int bfs(int stx, int sty, int end_x, int end_y) {
    vector<vector<int>> dist(n, vector<int>(m, -1));
    queue<pair<int, int>> q;
    dist[stx][sty] = 0;
    q.push({stx, sty});
    
    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        if (x == end_x && y == end_y) break;
        for (int i = 0; i < 4; i++) {
            int nx = x + dx[i], ny = y + dy[i];
            if (nx >= 0 && nx < n && ny >= 0 && ny < m && dist[nx][ny] == -1 && g[nx][ny] != '#') {
                dist[nx][ny] = dist[x][y] + 1;
                q.push({nx, ny});
            }
        }
    }
    return dist[end_x][end_y];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    for (int i = 0; i < n; i++) {
        cin >> g[i];
    }
    int xa, ya, xb, yb, xc, yc;
    cin >> xa >> ya >> xb >> yb >> xc >> yc;
    --xa; --ya; --xb; --yb; --xc; --yc;
    
    int first_ans = bfs(xa, ya, xb, yb);
    int second_ans = bfs(xb, yb, xc, yc);
    
    cout << first_ans + second_ans << endl;
    return 0;
}
