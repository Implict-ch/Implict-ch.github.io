# 题目内容

贫如洗的樵夫阿里巴巴在去砍柴的路上，无意中发现了强盗集团的藏宝地，藏宝地有编号从000~NNN的箱子每个箱子上面贴有一个数字箱子中可能有一个黄金宝箱。

黄金宝箱满足排在它之前的所有箱子数字和等于排在它之后的所有箱子数字之和;

一个箱子左边部分的数字和定义为000: 最后一个箱子右边部分的数字和定义为000.

帮阿里巴巴找到黄金宝箱，输出第一个满足条件的黄金宝箱编号，如果不存在黄金宝箱，请返回−1-1−1。

# 输入描述

箱子上贴的数字列表，使用逗号分隔，例如111，−1-1−1，000

宝箱的数量不小于111个，不超过100001000010000 宝箱上贴的数值范围不低于−1000-1000−1000，不超过100010001000

# 输出描述

第一个黄金宝箱的编号

# 样例

输入

2,5,-1,8,6

输出

3

说明

下标333之前的数字和为：2+5+−1=62+5+-1=62+5+−1=6

下标333之后的数字和为：6=66=66=6

输入

8,9

输出

-1

说明

不存在符合要求的位置

输入

11

输出

0

说明

下标000之前的数字和为：000

下标000之后的数字和为：000

# 思路：前缀和

给定nnn个数字，每个数字从前往后按照000~NNN进行编号。我们需要判断是否有一个编号，该编号左边所有数字和与该编号右边所有数字和相等。

在以上需求下，我们需要能够快速得到一段区间的和。常见的方法有树状数组、线段树、前缀和等。由于每个数字并不会在过程中发生变化，也就是说是静态的，因此我们选择前缀和的方法去求解。该方法能够在O(N)O(N)O(N)的预处理后，O(1)O(1)O(1)获得一段区间的和。

定义：

pre[i]pre[i]pre[i]表示∑aj,1≤j≤i\sum{a_j},1 \le j \le i∑a​j​​,1≤j≤i，那么有如下递推式：

pre[i]=pre[i−1]+a[i]pre[i]=pre[i-1]+a[i] pre[i]=pre[i−1]+a[i]

那么根据preprepre数组的定义，如果我们需要知道一段区间[l,r][l,r][l,r]的和，其结果就为：pre[r]−pre[l−1]pre[r]-pre[l-1]pre[r]−pre[l−1]。

# 代码

# C++

#include <iostream>
#include <cstdio>
using namespace std;

const int maxn=1e4+10;
int n;
int a[maxn];
int pre[maxn];

int main(){
	n++;
	scanf("%d",&a[n]);
	char c;
	while(scanf("%c",&c)){
		if(c!=',') break;
		n++;
		scanf("%d",&a[n]);
	}
	// 计算前缀和 
	for(int i=1;i<=n;++i){
		pre[i]=pre[i-1]+a[i];
	}
	// 枚举每一位的前后是否符合题意标准 
	for(int i=1;i<=n;++i){
		if(pre[i-1]==pre[n]-pre[i]){// 符合，则直接输出结果 
			printf("%d",i-1);
			return 0;
		}
	}
	// 不符合，输出-1表示无答案 
	printf("-1");
	
	return 0;
}

# Java

import java.util.Scanner;

public class Main {
    public static final int maxn = 10010;
    
    public static void main(String[] args) {
        int n = 0;
        int[] a = new int[maxn];
        int[] pre = new int[maxn];
        String s;
        
        Scanner scanner = new Scanner(System.in);
        
        s = scanner.nextLine();
        int number=0;
        boolean flag=false;
        for(int i=0;i<s.length();++i){
            if(s.charAt(i)==','){
                n++;
                if(flag==true){
                    a[n]=-number;
                }else{
                    a[n]=number;
                }
                number=0;
                flag=false;
            }else if(s.charAt(i)>='0'&&s.charAt(i)<='9'){
                number=number*10+s.charAt(i)-'0';
            }else if(s.charAt(i)=='-'){
                flag=true;
            }else break;
        }
        n++;
        if(flag==true){
            a[n]=-number;
        }else{
            a[n]=number;
        }
        
        // 计算前缀和
        for (int i = 1; i <= n; i++) {
            pre[i] = pre[i - 1] + a[i];
        }
        
        // 枚举每一位的前后是否符合题意标准
        for (int i = 1; i <= n; i++) {
            if (pre[i - 1] == pre[n] - pre[i]) { // 符合，则直接输出结果
                System.out.print(i - 1);
                return;
            }
        }
        
        // 不符合，输出-1表示无答案
        System.out.print("-1");
        
        
    }
}

		} else {
			break
		}

		a[n] = -number

const maxn = 10010;

function main() {
  const a = new Array(maxn).fill(0);
  const pre = new Array(maxn).fill(0);
  const readline = require("readline");

  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
  });

  rl.question("", (s) => {
    let n = 0;
    let number = 0;
    let flag = false;

    for (let i = 0; i < s.length; i++) {
      if (s[i] === ",") {
        n++;
        if (flag) {
          a[n] = -number;
        } else {
          a[n] = number;
        }
        number = 0;
        flag = false;
      } else if (s[i] >= "0" && s[i] <= "9") {
        number = number * 10 + parseInt(s[i]);
      } else if (s[i] === "-") {
        flag = true;
      } else {
        break;
      }
    }
    n++;
    if (flag) {
      a[n] = -number;
    } else {
      a[n] = number;
    }

    // 计算前缀和
    for (let i = 1; i <= n; i++) {
      pre[i] = pre[i - 1] + a[i];
    }

    // 枚举每一位的前后是否符合题意标准
    for (let i = 1; i <= n; i++) {
      if (pre[i - 1] === pre[n] - pre[i]) {
        // 符合，则直接输出结果
        console.log(i - 1);
        rl.close();
        return;
      }
    }

    // 不符合，输出-1表示无答案
    console.log("-1");
    rl.close();
  });
}

main();
