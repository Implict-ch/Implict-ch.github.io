# 题目内容

让我们来模拟一个消息队列 的运作，有一个发布者和若干消费者，发布者会在给定的时刻向消息队列发送消息,

  • 若此时消息队列有消费者订阅，这个消息会被发送到订阅的消费者中优先级最高(输入中消费者按优先级升序排列)的一个;
  • 若此时没有订阅的消费者，该消息被消息队列丢弃。

消费者则会在给定的时刻订阅消息队列或取消订阅。

  • 当消息发送和订阅发生在同一时刻时，先处理订阅操作，即同一时刻订阅的消费者成为消息发送的候选。
  • 当消息发送和取消订阅发生在同一时刻时，先处理取消订阅操作，即消息不会被发送到同一时刻取消订阅的消费者.

# 输入描述

输入为两行。 第一行为2N2N2N个正整数，代表发布者发送的NNN个消息的时刻和内容(为方便解折，消息内容也用正整教表示)。第一个数字是第一个消息的发送时刻，第二个数字是第一个消息的内容，以此类推。用例保证发送时刻不会重复，但注意消息并没有按照发送时刻排列。 第二行为2M2M2M个正整数，代表MMM个消费者订阅和取消订阅的时刻。第一个数字是第一个消费者订阅的时刻，第二个数字是第一个消费者取消订阅的时刻，以此类堆。用例保证每个消费者的取消订阅时刻大于订阅时刻，消费者按优先级 升席排列。

两行的数字都由空格分隔。NNN不超过100100100，MMM不超过101010，每行的长度不超过100010001000字符。

# 输出描述

输出为MMM行，依次为MMM个消费者收到的消息内容，消息内容按收到的顺序排列，且由空格分隔; 若某个消费者没有收到任何消息，则对应的行输出−1-1−1。

# 样例

输入

2 22 1 11 4 44 5 55 3 33
1 7 2 3

输出

11 33 44 55
22

说明

消息111111在111时刻到达，此时只有第一个消费者订阅，消息发送给它;

消息222222在222时刻到达，此时两个消费者都订阅了，消息发送给优先级最高的第二个消费者;

消息333333在时刻333到达，此时只有第一个消费者订阅，消息发送给它，

余下的消息按规则也是发送给第一个消费者。

输入

5 64 11 64 9 97
9 11 4 9

输出

97
64

说明

消息646464在555时刻到达，此时只有第二个消费者订阅，消息发送给它；

消息979797在999时刻到达，此时只有第一消费者订阅(因为第二个消费者刚好在999时刻取消订阅)，消息发送给它；

111111时刻也到达了一个内容为646464的消息，不过因为没有消费者订阅，消息被丢弃。

# 思路:模拟

本题阅读理解题，消费者优先级升席排序可以理解为读入后下标越大优先级越大

剩下的便是模拟题中的意思便可

# 代码

# C++

#include<bits/stdc++.h>

using namespace std;

int main() {
    vector<int> a;
    string s;
    getline(cin, s);
    stringstream ss; // 流读入
    ss << s;
    int num;
    while (ss >> num) {
        a.push_back(num);
    }
    
    vector<pair<int, int>> t;
    // 第一行输入两个一对，为发布者的发布时刻和发布内容
    for (int i = 0; i < (int)a.size(); i += 2) {
        t.push_back(make_pair(a[i], a[i + 1]));
    }
    // 按发布时间排序
    sort(t.begin(), t.end());
    
    vector<int> b;
    getline(cin, s);
    ss.clear();
    ss << s;
    while (ss >> num) {
        b.push_back(num);
    }
    
    vector<vector<int>> msg(b.size() / 2);
    
    for (auto i : t) {
        int x = i.first;
        int y = i.second;
        int ans = -1;
        // 从优先级高的订阅者开始遍历，优先匹配优先级高的订阅者
        for (int i = b.size() - 2; i >= 0; i -= 2) {
            if (b[i] <= x && x < b[i + 1]) { // 如果发布时刻在订阅时间中，订阅者收到发布内容
                ans = i / 2;
                break; // 只有优先级最高的收到
            }
        }
        if (ans != -1) {
            msg[ans].push_back(y); // 收到内容
        }
    }
    
    for (auto i : msg) {
        if (i.empty()) {
            cout << -1 << endl;
        } else {
            for (auto j : i) {
                cout << j << ' ';
            }
            cout << endl;
        }
    }

    return 0;
}

# Java

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        String[] aStr = scanner.nextLine().split(" ");
        int[] a = new int[aStr.length];
        for (int i = 0; i < aStr.length; i++) {
            a[i] = Integer.parseInt(aStr[i]);
        }

        List<Pair> t = new ArrayList<>();
        for (int i = 0; i < a.length; i += 2) { // 第一行输入两个一对，为发布者的发布时刻和发布内容
            t.add(new Pair(a[i], a[i + 1]));
        }
        t.sort(null); // 按发布时间排序

        String[] bStr = scanner.nextLine().split(" ");
        int[] b = new int[bStr.length];
        for (int i = 0; i < bStr.length; i++) {
            b[i] = Integer.parseInt(bStr[i]);
        }

        List<List<Integer>> msg = new ArrayList<>();
        for (int i = 0; i < b.length - 1; i += 2) {
            msg.add(new ArrayList<>());
        }

        for (Pair pair : t) {
            int x = pair.x;
            int y = pair.y;
            int ans = -1;
            for (int i = b.length - 2; i >= 0; i -= 2) { // 从优先级高的订阅者开始遍历，优先匹配优先级高的订阅者
                if (b[i] <= x && x < b[i + 1]) { // 如果发布时刻在订阅时间中，订阅者收到发布内容
                    ans = i / 2;
                    break; // 只有优先级最高的收到
                }
            }
            if (ans != -1) {
                msg.get(ans).add(y); // 收到内容
            }
        }

        for (List<Integer> list : msg) {
            if (list.isEmpty()) {
                System.out.println(-1);
            } else {
                for (int i : list) {
                    System.out.print(i + " ");
                }
                System.out.println();
            }
        }
    }

    static class Pair implements Comparable<Pair> {
        int x, y;

        public Pair(int x, int y) {
            this.x = x;
            this.y = y;
        }

        @Override
        public int compareTo(Pair other) {
            return Integer.compare(this.x, other.x);
        }
    }
}

  const t = [];
  for (let i = 0; i < a.length; i += 2) { // 第一行输入两个一对，为发布者的发布时刻和发布内容
    t.push([a[i], a[i + 1]]);
  }
  t.sort((x, y) => x[0] - y[0]); // 按发布时间排序

  rl.question('', (input) => {
    const b = input.split(' ').map(Number);
    const msg = Array.from({ length: b.length / 2 }, () => []);

    for (const [x, y] of t) {
      let ans = -1;
      for (let i = b.length - 2; i >= 0; i -= 2) { // 从优先级高的订阅者开始遍历，优先匹配优先级高的订阅者

          break; // 只有优先级最高的收到
        }
      }

        msg[ans].push(y); // 收到内容
      }

package main

import (
	"bufio"
	"fmt"
	"os"
	"sort"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)

	// 获取输入
	s, _ := reader.ReadString('\n')
	s = strings.TrimSpace(s)
	ss := strings.Split(s, " ")
	var a []int
	for _, numStr := range ss {
		num, _ := strconv.Atoi(numStr)
		a = append(a, num)
	}

	var t []pair
	for i := 0; i < len(a); i += 2 { // 第一行输入两个一对，为发布者的发布时刻和发布内容
		t = append(t, pair{a[i], a[i+1]})
	}
	sort.Slice(t, func(i, j int) bool { // 按发布时间排序
		return t[i].first < t[j].first
	})

	// 获取输入
	s, _ = reader.ReadString('\n')
	s = strings.TrimSpace(s)
	ss = strings.Split(s, " ")
	var b []int
	for _, numStr := range ss {
		num, _ := strconv.Atoi(numStr)
		b = append(b, num)
	}

	msg := make([][]int, len(b)/2)
	for _, p := range t {
		x := p.first
		y := p.second
		ans := -1
		for i := len(b) - 2; i >= 0; i -= 2 { // 从优先级高的订阅者开始遍历，优先匹配优先级高的订阅者
			if b[i] <= x && x < b[i+1] { // 如果发布时刻在订阅时间中，订阅者收到发布内容
				ans = i / 2
				break // 只有优先级最高的收到
			}
		}
		if ans != -1 {
			msg[ans] = append(msg[ans], y) // 收到内容
		}
	}

	// 输出结果
	for _, m := range msg {
		if len(m) == 0 {
			fmt.Println(-1)
		} else {
			for _, j := range m {
				fmt.Printf("%d ", j)
			}
			fmt.Println()
		}
	}
}
