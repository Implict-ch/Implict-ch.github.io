# 题目描述

小明和朋友玩跳格子游戏，有nnn个连续格子，每个格子有不同的分数，小朋友可以选择以任意格子起跳，但是不能跳连续的格子，也不能回头跳;

给定一个代表每个格子得分的非负整数数组，计算能够得到的最高分数。

# 输入描述

给定一个数列，如:111222333111

# 输出描述

输出能够得到的最高分，如:444

备注

1≤nums.length≤1001 \leq nums.length \leq 1001≤nums.length≤100

0≤numsli]≤10000 \leq numsli] \leq 10000≤numsli]≤1000

# 样例

输入

1 2 3 1

输出

4

说明

选择跳第一个格子和第三个格子

输入

2 7 9 3 1

输出

12

说明

2+9+1=122+9+1=122+9+1=12

# 思路:dp

设dp[i]表示前i个格子可以取得的最大数值

若前i个只跳到第i-1格，则

dp[i] = dp[i - 1]

若跳到第i格，则

dp[i] = dp[i - 2] + a[i]

两者取最大

最终输出最大的dp值既是答案

# 代码

# C++

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    vector<int> a;
    int num;
    char delimiter;
    while (cin >> num) {
        a.push_back(num);
    }
	
    vector<int> d(a.size(), 0);
    d[0] = a[0]; // 初始化
    if (a.size() > 1) {
        d[1] = max(a[0], a[1]);
    }

    for (int i = 2; i < a.size(); i++) {
        d[i] = max(d[i - 1], d[i - 2] + a[i]); // 状态转移
    }

    cout << *max_element(d.begin(), d.end()) << endl; // 找到最大的dp值

    return 0;
}

# Go

package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)

	var a []int

	for scanner.Scan() {
		num, _ := strconv.Atoi(scanner.Text())
		a = append(a, num)
	}

	d := make([]int, len(a))
	d[0] = a[0] // 初始化
	if len(a) > 1 {
		d[1] = max(a[0], a[1])
	}

	for i := 2; i < len(a); i++ {
		d[i] = max(d[i-1], d[i-2]+a[i]) // 状态转移
	}

	fmt.Println(maxElement(d))// 找到最大的dp值
}

func max(a, b int) int {
	if a > b {
		return a
	}
	return b
}

func maxElement(a []int) int {
	max := a[0]
	for _, num := range a {
		if num > max {
			max = num
		}
	}
	return max
}
