# 题目描述

对报文进行重传和重排序是常用的可靠性机制，重传缓中区内有一定数量的子报文，每个子报文在原始报文中的顺序已知，现在需要恢复出原始报文。

# 输入描述

输入第一行为NNN，表示子报文的个数，0<N≤10000 <N \leq 10000<N≤1000。

输入第二行为NNN个子报文，以空格分开，子报文格式为:

字符串报文内容+后缀顺序索引

字符串报文内容由[a−z,A−Z][a-z,A-Z][a−z,A−Z]组成，后缀为整型值，表示顺序。 顺序值唯一，不重复。

# 输出描述

输出恢复出的原始报文，按照每个子报文的顺序的升序排序恢复出原始报文，顺序后缀需要从恢复出的报文中删除掉

# 样例

输入

4
rolling3  stone4  like1  a2

输出

like  a  rolling  stone

说明

444个子报文的内容分别为"rollingrollingrolling”，"stonestonestone”，"likelikelike”，"aaa"，顺序值分别为3，4，1，2，按照顺序值升序并删除顺序后缀，得到恢复的原始报文:"likelikelike aaa rollingrollingrolling stonestonestone“

输入

8
gifts6  and7  Exchanging1  all2  precious5  thins8  kinds3  of4

输出

Exchanging  all  kinds  of  precious  gifts  and  things

说明

无

# 思路：模拟

给定N个字符串，每个字符串格式为：

字符串报文内容+后缀顺序索引

我们需要按照后缀顺序索引排序后，将字符串报文内容输出。

其实考察的是处理字符串的基本功。

由于题目里保证了字符串报文内容不会出现数字，因此我们只需要在读到数字(假定字符为c，那么′0′≤c≤′9′'0'\le c \le '9'​′​​0​′​​≤c≤​′​​9​′​​)时记录数字(其后均为数字，直到读到空格或者回车为止)，然后标记为当前字符串的编号即可。

所有串按编号排序后输出。

# 代码

# C++

#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n;

struct node{
	char s[1010];
	int len;
	int number;
}a[1010];

bool cmp(node a,node b){
	return a.number<b.number;
}

int main(){
	scanf("%d",&n);
	node tmp;
	for(int i=1,number,len;i<=n;++i){
		scanf("%s",tmp.s);
		number=0;
		len=strlen(tmp.s);
		for(int j=0;j<len;++j){
			if(tmp.s[j]>='0'&&tmp.s[j]<='9'){// 查找当前串的数字 
				number=number*10+tmp.s[j]-'0';
			}
		}
		for(int j=0;j<len;++j){// 存储当前串的字符串 
			if(tmp.s[j]>='0'&&tmp.s[j]<='9'){
				break;
			}
			a[i].s[j]=tmp.s[j];
		}
		a[i].len=strlen(a[i].s);
		a[i].number=number;
	}
	// 排序输出 
	sort(a+1,a+n+1,cmp);
	for(int i=1;i<=n;++i){
		for(int j=0;j<a[i].len;++j){
			printf("%c",a[i].s[j]);
		}
		printf(" ");
	}
	
	
	return 0;
}

# Java

import java.util.Arrays;
import java.util.Scanner;

class Main {
    static int n;

    static class Node {
        String s;  // 存储当前串的字符串
        int len;  // 当前串的长度
        int number;  // 当前串的数字
    }

    static Node[] a;

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        n = scanner.nextInt();  // 读取输入的整数 n
        a = new Node[n + 1];  // 创建节点数组，长度为 n+1，索引从 1 开始

        for (int i = 1; i <= n; i++) {
            Node tmp = new Node();
            tmp.s = scanner.next();  // 读取下一个字符串
            int number = 0;
            int len = tmp.s.length();
            for (int j = 0; j < len; j++) {
                if (tmp.s.charAt(j) >= '0' && tmp.s.charAt(j) <= '9') {
                    number = number * 10 + (tmp.s.charAt(j) - '0');  // 计算当前串的数字部分
                }
            }
            StringBuilder sb = new StringBuilder();
            for (int j = 0; j < len; j++) {
                if (tmp.s.charAt(j) >= '0' && tmp.s.charAt(j) <= '9') {
                    break;
                }
                sb.append(tmp.s.charAt(j));  // 获取当前串的非数字部分
            }
            tmp.len = sb.length();  // 非数字部分的长度
            tmp.number = number;  // 当前串的数字部分
            a[i] = tmp;  // 将节点存储到数组中
        }

        Arrays.sort(a, 1, n + 1, (a, b) -> Integer.compare(a.number, b.number));  // 按照 number 属性对 a 数组进行排序

        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < a[i].len; j++) {
                System.out.print(a[i].s.charAt(j));  // 输出当前节点的字符串部分
            }
            System.out.print(" ");  // 输出空格分隔符
        }

        scanner.close();
    }
}

# Go

package main

import (
	"fmt"
	"sort"
	"strings"
)

type Node struct {
	s      string // 存储当前串的字符串
	len    int    // 当前串的长度
	number int    // 当前串的数字
}

func main() {
	var n int
	fmt.Scan(&n) // 读取输入的整数 n

	a := make([]Node, n)

	for i := 0; i < n; i++ {
		var str string
		fmt.Scan(&str) // 读取下一个字符串

		number := 0
		len := len(str)
		for j := 0; j < len; j++ {
			if str[j] >= '0' && str[j] <= '9' {
				number = number*10 + int(str[j]-'0') // 计算当前串的数字部分
			}
		}

		var sb strings.Builder
		for j := 0; j < len; j++ {
			if str[j] >= '0' && str[j] <= '9' {
				break
			}
			sb.WriteByte(str[j]) // 获取当前串的非数字部分
		}

		tmp := Node{
			s:      sb.String(), // 非数字部分的字符串
			len:    sb.Len(),    // 非数字部分的长度
			number: number,      // 当前串的数字部分
		}

		a[i] = tmp // 将节点存储到数组中
	}

	sort.Slice(a, func(i, j int) bool {
		return a[i].number < a[j].number // 按照 number 属性对 a 数组进行排序
	})

	for i := 0; i < n; i++ {
		fmt.Print(a[i].s) // 输出当前节点的字符串部分
		fmt.Print(" ")    // 输出空格分隔符
	}
}

# Js

const readline = require('readline');

const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

function cmp(a, b) {
  return a.number - b.number;
}

function main() {
  let n; // 用于存储要读取的字符串数量
  const a = []; // 用于存储节点（带有数字的字符串）

  rl.on('line', (input) => { // 监听读取输入行的事件
    if (!n) { // 如果n还未初始化，表示正在读取第一行包含字符串数量的输入
      n = parseInt(input); // 解析输入为整数，获取字符串数量
    } else { // 否则，正在读取字符串本身
      const str = input.trim(); // 去除字符串两端的空白字符
      let len = str.length; // 字符串长度
      for(let i=1,j=0;i<=n;++i,++j){
          let number = 0; // 当前字符串的数字部分
          const tmp = {
            s: '', // 当前字符串的非数字部分
            len: 0, // 当前字符串的长度
            number: 0 // 当前字符串的数字部分
          };
          for(;j<len;++j){
              if(str[j]==' '){ // 遇到空格时判断是否为多余的空格
                  if(str[j+1]==' ') continue; // 如果是多余的空格则跳过
                  else break; // 否则表示当前字符串已读取完毕
              }
              if(str[j]>='0'&&str[j]<='9'){ // 查找当前字符串的数字部分
                  number=number*10+parseInt(str[j]);
              }else{
                  tmp.s+=str[j]; // 获取当前字符串的非数字部分
              }
          }
          tmp.len = tmp.s.length; // 非数字部分的长度
          tmp.number = number; // 当前字符串的数字部分
          a.push(tmp); // 将节点存储到数组中
      }
      if (a.length === n) { // 如果节点数组的长度等于字符串数量，表示所有字符串已读取完毕
        rl.close(); // 关闭读取接口

        a.sort(cmp); // 根据number属性对节点数组进行排序

        let result = '';
        for (let i = 0; i < n; i++) {
          result += a[i].s + ' '; // 输出当前节点的字符串部分
        }

        console.log(result.trim()); // 打印结果，去除末尾的空格
      }
    }
  });
}

main();
