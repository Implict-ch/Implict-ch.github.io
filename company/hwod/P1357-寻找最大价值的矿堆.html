# 题目描述

给你一个由'000'(空地)、'111'(银矿)、'222'金矿)组成的的地图，矿堆只能由上下左右相邻的金矿或银矿连接形成。超出地图范围可以认为是空地。

假设银矿价值111，金矿价值222，请你找出地图中最大价值的矿堆并输出该矿堆的价值.

# 输入描述

地图元素信息如:

22220
00000
00000
11111
  • 地图范围最大300∗300300*300300∗300 -0 \leq 地图元素 \leq 2

# 输出描述

矿堆的最大价值

# 样例

输入

22220
00000
00000
01111

输出

8

说明

无

输入

22220
00020
00010
01111

输出

15

说明

无

输入

20000
00020
00000
00111

输出

3

说明

无

# 思路

首先，我们发现题目的意思就是求所有矿堆的最大权值，然后我们发现矿堆这一个概念和图论中的联通块概念一样。因此这道题本质就是去求一张图中权值最大的联通块。而所谓的图便是题目给的包含012三个矿石的无向图（因为任意相邻两点都可以相互到达，所以是无向图）。

# 如何建图：

我们可以把0权值的点当作不可到达的点，即任何点都不存在与0权值节点的边。其他权值的点由于存在上下左右四个方向，因此，对于四个方向权值非0的点连接一条边。最后，按照这个建图方式即可建立一整张图。然而我们发现，题目给的数据其实我们已经可以认为建图完成了，只要后续遍历遇到权值为0的点跳过即可。

# 无向图求联通块思路：

无向图求联通块我们可以使用dfsdfsdfs。我们可以想象一下，对于无向图，一次dfsdfsdfs其实就正好遍历完一个联通块，因此，为了求得所有的联通块权值，我们就需要对所有联通块进行dfsdfsdfs。

在dfsdfsdfs的过程中，如何求得联通块的权值？很明显，我们只要知道遍历了哪些点，然后统计总权值即可。因此，dfs的目的只是在于遍历联通块所有的点。然而，图上dfsdfsdfs存在一个问题，就是一个点可能被多次dfsdfsdfs到(比如点(x,y)(x,y)(x,y)可以dfsdfsdfs到(x+1,y)(x+1,y)(x+1,y),点(x+2,y)(x+2,y)(x+2,y)也可以dfsdfsdfs到(x+1,y)(x+1,y)(x+1,y))，因此，我们需要改善我们的搜索策略，对于已经遍历过的点直接跳过(比如点(x,y)(x,y)(x,y)可以dfsdfsdfs到(x+1,y)(x+1,y)(x+1,y)过后,下次点(x+2,y)(x+2,y)(x+2,y)也dfsdfsdfs到(x+1,y)(x+1,y)(x+1,y)时直接returnreturnreturn)，这样，我们可以保证一张连通图的所有点都只会被遍历一次，与此同时，由于只遍历一次，我们求总权值就可以直接将所有的权值加起来即可。

# 代码

# C++

#include <iostream>
#include <vector>
using namespace std;

vector<vector<int>> arr;
int n, m;

int dfs(int x, int y) {
    if (arr[x][y] == 0) {//如果遇到走不通的路，就跳过
        return 0;
    }
    int ans = arr[x][y];//记录一个联通块的所有权值和
    arr[x][y] = 0;//通过这个操作，并结合上面的return，可以使得一个联通块的有权值的点只会被统计一次
    vector<pair<int, int>> dir = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};//四个方向
    for (auto d : dir) {
        int fx = d.first, fy = d.second;
        int xx = fx + x, yy = fy + y;//目标点(xx,yy)
        if (0 <= xx && xx < n && 0 <= yy && yy < m) {//合法
            ans += dfs(xx, yy);//就进行dfs，统计点(xx,yy)的权值，我们发现，如果(xx,yy)的权值已经被记录到ans里，那么dfs(xx,yy)的值等于0，这配合return就可以实现ans=sum(联通块)
        }
    }
    return ans;
}

int main() {
    string line;
    while(cin >> line){
        vector<int> ls;
        for (char c : line) {
            ls.push_back(c - '0');
        }
        arr.push_back(ls);
    }//二维矩阵建图，图里存的都是整数0，1，2

    n = arr.size();
    m = arr[0].size();
    int res = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            res = max(res, dfs(i, j));//找联通块，并且查找最大值
        }
    }

    cout << res << endl;
    return 0;
}

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Main {
    static List<List<Integer>> arr;
    static int n, m;

    static int dfs(int x, int y) {
        if (arr.get(x).get(y) == 0) {//如果遇到走不通的路，就跳过
            return 0;
        }
        int ans = arr.get(x).get(y);//记录一个联通块的所有权值和
        arr.get(x).set(y, 0);//通过这个操作，并结合上面的return，可以使得一个联通块的有权值的点只会被统计一次
        int[][] dir = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};//四个方向
        for (int[] d : dir) {
            int fx = d[0], fy = d[1];
            int xx = fx + x, yy = fy + y;//目标点(xx,yy)
            if (0 <= xx && xx < n && 0 <= yy && yy < m) {//合法
                ans += dfs(xx, yy);//就进行dfs，统计点(xx,yy)的权值，我们发现，如果(xx,yy)的权值已经被记录到ans里，那么dfs(xx,yy)的值等于0，这配合return就可以实现ans=sum(联通块)
            }
        }
        return ans;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        arr = new ArrayList<>();
        while (scanner.hasNextLine()) {
            String line = scanner.nextLine().trim();
            if (line.length() == 0) {
                continue;
            }
            List<Integer> ls = new ArrayList<>();
            for (char c : line.toCharArray()) {
                ls.add(c - '0');
            }
            arr.add(ls);//二维矩阵建图，图里存的都是整数0，1，2
        }

        n = arr.size();
        m = arr.get(0).size();
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                res = Math.max(res, dfs(i, j));//找联通块，并且查找最大值
            }
        }

        System.out.println(res);
    }
}

// 记录地图矩阵的行数row，列数col
let row, col;
 
function getResult(matrix) {
  // 记录地图矩阵的行数row
  row = matrix.length;
  if (row == 0) return 0;
 
  // 记录地图矩阵的列数col
  col = matrix[0].length;
 
  // 记录最大矿堆价值
  let ans = 0;
 
  // 遍历矩阵元素
  for (let i = 0; i < row; i++) {
    for (let j = 0; j < col; j++) {
      if (matrix[i][j] > 0) {
        ans = Math.max(ans, dfs(i, j));
      }
    }
  }
 
  return ans;
}
 
// 上下左右，四个方向的偏移量
const offsets = [
  [-1, 0],
  [1, 0],
  [0, -1],
  [0, 1],
];
 
function dfs(x, y) {
  const stack = [[x, y]];
  let sum = 0;
 
  while (stack.length > 0) {
    const [x, y] = stack.pop();
 
    sum += matrix[x][y];
    matrix[x][y] = 0;
 
    for (let offset of offsets) {
      const newX = x + offset[0];
      const newY = y + offset[1];
 
      if (
        newX >= 0 &&
        newX < row &&
        newY >= 0 &&
        newY < col &&
        matrix[newX][newY] > 0
      ) {
        stack.push([newX, newY]);
      }
    }
  }
 
  return sum;
}
