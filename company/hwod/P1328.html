# 题目描述

一贫如洗的樵夫阿里巴巴在去砍柴的路上，无意中发现了强盗集团的藏宝地，藏宝地有编号从0−N0-N0−N的箱子，每个箱子上面贴有箱子中藏有金币QQQ的数量。 从金币数量中选出一个数字集合，并销毁贴有这些数字的每个箱子，如果能销毁一半及以上的箱子，则返回这个数字集合的最小大小

# 输入描述

第一行1个数字字串，数字之间使用逗号分隔，例如:6,6,6,6,3,3,3,1,1,56,6,6,6,3,3,3,1,1,56,6,6,6,3,3,3,1,1,5字串中数字的个数为偶数，并且

  • 1≤1\leq1≤字串中数字的个数≤100000\leq 100000≤100000
  • 1≤1\leq1≤每个数字 ≤100000\leq 100000≤100000

# 输出描述

这个数字集合的最小大小

# 样例

输入

1,1,1,1,3,3,3,6,6,8

输出

2

说明

选择集合1,8{1,8}1,8，销毁后的结果数组为[3,3,3,6,6][3,3,3,6,6][3,3,3,6,6]，长度为555，长度为原数组的一半。

大小为 222 的可行集合还有{1,31,31,3},{1,61,61,6},{3,63,63,6}。

选择6,8{6,8}6,8集合是不可行的，它销后的结果数组为[1,1,1,1,3,3,3][1,1,1,1,3,3,3][1,1,1,1,3,3,3]，新数组长度大于原数组的二分之一。

输入

2,2,2,2

输出

1

说明

我们只能选择集合{222}，销毁后的结果数组为空。

# 思路：贪心

给定nnn个数字，我们在选择一个数字后，会将所有该数字都消除。现在要求剩余数字个数不超过原来的一半，求出所选数字集合的最小大小。

观察题意可以知道，我们选一个数字会消除所有该数字，而最后又要求剩余数字的个数不超过原来的一半。那么理所当然的，我们就需要统计每一个数有多少个。

统计之后呢？怎样才能让该集合最小？一个很直观的想法就是，每次都删除个数最多的数字，这样剩余数字的数量也会减少的最快。这就是本题的贪心思路。

于是，我们在统计完后，对该统计个数集合进行排序，并从大的开始删，直到达成目标即可。

# 代码

# C++

#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

const int maxn=1e5+10;
int n,m;
int a[maxn],b[maxn];
string s;

int main(){
	cin>>s;
	int number=0;
	for(int i=0;i<s.length();++i){
		if(s[i]>='0'&&s[i]<='9'){
			number=number*10+s[i]-'0';
		}else{
			n++;
			a[n]=number;
			number=0;
		}
	}
	n++;
	a[n]=number;
	sort(a+1,a+n+1);
	// 对a数组进行计数后，存储在b数组中 
	int cnt=1;
	for(int i=2;i<=n+1;++i){
		if(a[i]==a[i-1]) cnt++;
		else{
			b[++m]=cnt;
			cnt=1;
		}
	}
	b[++m]=cnt; 
	sort(b+1,b+m+1);
	// 对b数组排序，从计数多的数开始删 
	for(int i=m,tmp=0;i;--i){
		tmp+=b[i];
		if(tmp>=n/2){
			printf("%d",m-i+1);
			break;
		}
	}
	
	
	return 0;
}

# Java

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        List<Integer> a = new ArrayList<>();
        List<Integer> b = new ArrayList<>();
        int n = 0;
        int m = 0;

        Scanner scanner = new Scanner(System.in);
        String[] s = scanner.nextLine().split(",");

        n = s.length;
        a.add(0);
        for (int i = 0; i < n; i++) {
            a.add(Integer.parseInt(s[i]));
        }
        // 对a数组进行排序
        a.sort(null);
        
        // 对a数组进行计数后，存储在b数组中
        int cnt = 1;
        for (int i = 2; i <= n; i++) {
            if (a.get(i).equals(a.get(i - 1))) {
                cnt++;
            } else {
                b.add(cnt);
                cnt = 1;
            }
        }
        b.add(cnt);
        // 对b数组排序，从计数多的数开始删
        Integer[] bArray = b.toArray(new Integer[0]);
        Arrays.sort(bArray);
        m = bArray.length;
        int tmp = 0;
        for (int i = m - 1; i > 0; i--) {
            tmp += bArray[i];
            if (tmp >= n / 2) {
                System.out.println(m - i);
                break;
            }
        }
    }
}

package main

import (
	"fmt"
	"sort"
	"strconv"
	"strings"
)

func main() {
	var a []int
	var b []int
	var n, m int

	var input string
	fmt.Scanln(&input)
	s := strings.Split(input, ",")

	n = len(s)
	a = append(a, 0)
	for i := 0; i < n; i++ {
		num, _ := strconv.Atoi(s[i])
		a = append(a, num)
	}
	// 对a数组进行排序
	sort.Ints(a)

	// 对a数组进行计数后，存储在b数组中
	cnt := 1
	for i := 2; i <= n; i++ {
		if a[i] == a[i-1] {
			cnt++
		} else {
			b = append(b, cnt)
			cnt = 1
		}
	}
	b = append(b, cnt)
	// 对b数组排序，从计数多的数开始删
	sort.Ints(b)
	m = len(b)
	tmp := 0
	for i := m - 1; i > 0; i-- {
		tmp += b[i]
		if tmp >= n/2 {
			fmt.Println(m - i)

		}
	}
}

# Js

const readline = require('readline');

const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

function main() {
  let a = [];
  let b = [];
  let n, m;

  rl.question('', (input) => {
    let s = input.split(',');

    n = s.length;
    a.push(0);
    for (let i = 0; i < n; i++) {
      let num = parseInt(s[i]);
      a.push(num);
    }

    // 对a数组进行排序
    a.sort((x, y) => x - y);

    // 对a数组进行计数后，存储在b数组中
    let cnt = 1;
    for (let i = 2; i <= n; i++) {
      if (a[i] === a[i - 1]) {
        cnt++;
      } else {
        b.push(cnt);
        cnt = 1;
      }
    }
    b.push(cnt);

    // 对b数组排序，从计数多的数开始删
    b.sort((x, y) => x - y);
    m = b.length;
    let tmp = 0;
    for (let i = m - 1; i > 0; i--) {
      tmp += b[i];
      if (tmp >= n / 2) {
        console.log(m - i);
        rl.close();
        return;
      }
    }

    rl.close();
  });
}

main();
