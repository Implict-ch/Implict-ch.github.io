# 题目描述

贫如洗的樵夫阿里巴巴在去砍柴的路上，无意中发现了强盗集团的藏宝地，藏宝地有编号从0−N0-N0−N的箱子，每个箱子上面贴有一个数字。

阿里巴巴念出一个咒语数字，查看宝箱是否存在两个不同箱子，这两个箱子上贴的数字相同，同时这两个箱了的编号之差的绝对值小于等于咒语数字，

如果存在这样的一对宝箱，请返回最先找到的那对宝箱左边箱子的编号，如果不存在则返回−1-1−1

# 输入描述

第一行输入一个数字字串，数字之间使用逗号分隔，例如: 1,2,3,11,2,3,11,2,3,1

  • 1≤1\leq1≤字串中数字个数 ≤100000\leq100000≤100000
  • −100000≤-100000\leq−100000≤每个数字值≤100000\leq100000≤100000

第二行输入咒语数字，例如:333

  • 1≤1\leq1≤ 咒语数字 ≤100000\leq100000≤100000

# 输出描述

存在这样的一对宝箱，请返回最先找到的那对宝箱左边箱子的编号，如果不存在则返回−1-1−1

# 样例

输入

6,3,1,6
3

输出

0

说明

无

输入

5,6,7,5,6,7
2

输出

-1

说明

无

# 思路：遍历

给定NNN个箱子，每个箱子有一个值，我们需要找到是否有箱子值相同，并且箱子的编号差的绝对值小于给定值。

读完题意，一个问题跃然纸上：我们要怎样快速找到和当前数相同的下一个数的位置呢？

我们可以采用标记法逆序遍历数组，从而得到我们想要的目标。

定义：

数组：nxt[i]nxt[i]nxt[i]表示与第iii个数相同的下一个数的位置;last[i]last[i]last[i]表示数字iii上次出现的下标

nxtnxtnxt数组即为我们想要的数组，而lastlastlast数组则是用来辅助我们更新nxtnxtnxt的数组。

那么我们就可以从大往小枚举，按照两个数组的定义不断更新数组。

如果你看了数组定义仍旧想不到更新方法的话，可以结合代码理解。这是一个很常见的标记的方法，应该牢记。

需要注意的是，由于数字存在负数，因此我们可以对每个数加一个偏置值，从而将负数变成正数，之后继续后面的操作。

之后，我们再从前往后遍历，检查是否有符合题意得答案即可。

# 代码

# C++

#include <iostream>
#include <cstdio>
using namespace std;

const int maxn=1e5+10;
int a[maxn],k,n;
int nxt[maxn],last[maxn<<1];
string s;

int main(){
	cin>>s;
	int number=0;
	bool op=false;
	for(int i=0;i<s.length();++i){
		if(s[i]>='0'&&s[i]<='9'){
			number=number*10+s[i]-'0';
		}else if(s[i]=='-'){
			op=true;
		}else{
			n++;
			a[n]=number;
			if(op) a[n]=-a[n];
			number=0;
			op=false;
		}
	}
	n++;
	a[n]=number;
	if(op) a[n]=-a[n];
	scanf("%d",&k);
	for(int i=n;i;--i){// 从大往小枚举 
		if(last[maxn+a[i]]==0){// 如果上一个位置是0，说明是该数第一次出现 
			nxt[i]=n+1;// 标记为n+1 
		}else{
			nxt[i]=last[maxn+a[i]];// 标记为该数上一次出现的位置 
		}
		last[maxn+a[i]]=i;// 更新该数上一次出现的位置 
	}
	for(int i=1;i<=n;++i){
		if(nxt[i]!=n+1&&nxt[i]-i<=k){// 判断是否有数符合条件 
			printf("%d",i-1);
			return 0;
		}
	}
	printf("-1");
	
	return 0;
}

# python

maxn = 10**5 + 10  # 定义最大值
a = [0] * maxn  # 数组a，用于存储输入的数字序列
nxt = [0] * maxn  # 数组nxt，用于存储每个数字的下一个出现位置
last = [0] * (maxn * 2)  # 数组last，用于存储每个数字的上一个出现位置

s = input().strip().split(',')  # 输入数字序列并按逗号分隔

n = len(s)  # 数字序列的长度
for i in range(len(s)):
    a[i+1] = int(s[i])  # 将数字序列转换为整数并存入数组a

k = int(input())  # 输入k值

for i in range(n, 0, -1):  # 从大到小遍历数字序列
    if last[maxn + a[i]] == 0:  # 如果该数字上一次出现位置为0，说明是第一次出现
        nxt[i] = n + 1  # 标记为n+1
    else:
        nxt[i] = last[maxn + a[i]]  # 标记为该数字上一次出现的位置
    last[maxn + a[i]] = i  # 更新该数字上一次出现的位置

for i in range(1, n + 1):  # 遍历数字序列
    if nxt[i] != n + 1 and nxt[i] - i <= k:  # 判断是否存在满足条件的数字
        print(i - 1)
        exit(0)

print("-1")  # 未找到符合条件的数字，输出-1

# Java

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int maxn = 100010;  // 定义最大值
        int[] a = new int[maxn];  // 数组a，用于存储输入的数字序列
        int[] nxt = new int[maxn];  // 数组nxt，用于存储每个数字的下一个出现位置
        int[] last = new int[maxn << 1];  // 数组last，用于存储每个数字的上一个出现位置

        String s = scanner.next();  // 输入数字序列
        int number = 0;  // 临时变量，用于构建数字
        boolean op = false;  // 是否为负数的标志
        int n = 0;  // 数字序列的长度
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c >= '0' && c <= '9') {  // 如果是数字字符
                number = number * 10 + (c - '0');  // 构建数字
            } else if (c == '-') {  // 如果是负号
                op = true;  // 标记为负数
            } else {  // 如果是分隔符
                n++;  // 增加数字序列长度
                a[n] = number;  // 存储数字到数组a
                if (op) {  // 如果是负数
                    a[n] = -a[n];  // 取负数
                }
                number = 0;  // 重置临时变量
                op = false;  // 重置标志
            }
        }
        n++;  // 增加数字序列长度，存储最后一个数字
        a[n] = number;  // 存储数字到数组a
        if (op) {  // 如果是负数
            a[n] = -a[n];  // 取负数
        }

        int k = scanner.nextInt();  // 输入k值

        for (int i = n; i >= 1; i--) {  // 从大往小枚举
            if (last[maxn + a[i]] == 0) {  // 如果上一个位置是0，说明是该数第一次出现
                nxt[i] = n + 1;  // 标记为n+1
            } else {
                nxt[i] = last[maxn + a[i]];  // 标记为该数上一次出现的位置
            }
            last[maxn + a[i]] = i;  // 更新该数上一次出现的位置
        }

        for (int i = 1; i <= n; i++) {
            if (nxt[i] != n + 1 && nxt[i] - i <= k) {  // 判断是否有数符合条件
                System.out.print(i - 1);
                return;
            }
        }

        System.out.print("-1");  // 未找到符合条件的数字，输出-1
    }
}

package main

import (
	"fmt"
	"strconv"
)

func main() {
	const maxn = 100010
	var a [maxn]int    // 存储输入的数字序列
	var nxt [maxn]int  // 存储每个数字的下一个出现位置
	var last [maxn * 2]int  // 存储每个数字的上一个出现位置

	var s string
	fmt.Scan(&s)  // 输入数字序列

	var number = 0
	var op = false
	var n = 0
	for i := 0; i < len(s); i++ {
		c := s[i]
		if c >= '0' && c <= '9' {
			digit, _ := strconv.Atoi(string(c))
			number = number*10 + digit
		} else if c == '-' {
			op = true
		} else {
			n++
			a[n] = number
			if op {
				a[n] = -a[n]
			}
			number = 0
			op = false
		}
	}
	n++
	a[n] = number
	if op {
		a[n] = -a[n]
	}

	var k int
	fmt.Scan(&k)  // 输入k值

	for i := n; i >= 1; i-- {  // 从大往小枚举
		if last[maxn+a[i]] == 0 {  // 如果上一个位置是0，说明是该数第一次出现
			nxt[i] = n + 1  // 标记为n+1
		} else {
			nxt[i] = last[maxn+a[i]]  // 标记为该数上一次出现的位置
		}
		last[maxn+a[i]] = i  // 更新该数上一次出现的位置
	}

	for i := 1; i <= n; i++ {
		if nxt[i] != n+1 && nxt[i]-i <= k {  // 判断是否有数符合条件
			fmt.Print(i - 1)
			return
		}
	}

	fmt.Print("-1")  // 未找到符合条件的数字，输出-1
}

# Js

const readline = require('readline');

const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

const maxn = 100010;
const a = new Array(maxn).fill(0);
const nxt = new Array(maxn).fill(0);
const last = new Array(maxn * 2).fill(0);

rl.on('line', (input) => {
  const tokens = input.split(' ');

  if (tokens.length === 1) {
    let s = tokens[0];
    let number = 0;
    let op = false;
    let n = 0;

    for (let i = 0; i < s.length; i++) {
      const c = s[i];
      if (c >= '0' && c <= '9') {
        number = number * 10 + parseInt(c);
      } else if (c === '-') {
        op = true;
      } else {
        n++;
        a[n] = number;
        if (op) {
          a[n] = -a[n];
        }
        number = 0;
        op = false;
      }
    }
    n++;
    a[n] = number;
    if (op) {
      a[n] = -a[n];
    }

    rl.on('line', (input) => {
      const k = parseInt(input);

      for (let i = n; i >= 1; i--) {
        if (last[maxn + a[i]] === 0) {
          nxt[i] = n + 1;
        } else {
          nxt[i] = last[maxn + a[i]];
        }
        last[maxn + a[i]] = i;
      }

      for (let i = 1; i <= n; i++) {
        if (nxt[i] !== n + 1 && nxt[i] - i <= k) {
          console.log(i - 1);
          return;
        }
      }

      console.log("-1");
    });
  }
});
