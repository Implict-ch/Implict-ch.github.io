# 思路:双指针

首先抓住题干一点：保证输入可以还原成唯一序列

然后本题便可考虑是用双指针

具体做法为：

  1. 记录给定字符串的0~9的出现次数，用数组a记录，为了方便解释，我们将给定的字符串记作S
  2. 设置两个左右指针，初始都为1
  3. 右指针一直递增，并用数组b记录生成字符串的0至9的出现次数（即此时左右指针范围内的连续数组成的字符串的0~9的出现次数，为了方便我们记作T）
  4. 若b数组中的任意一个下标的个数大于a数组对应下标的个数时（即T的某个数字个数出现次数超过了S对应的数字的出现次数），左指针递增，并更新b数组（因为左指针递增意味着T少了一些数字）
  5. 重复3~5步骤，直到a数组等于b数组，并且len(S) == len(T)，左指针即为答案

# 代码

# C++

#include <bits/stdc++.h>

using namespace std;

int main() {
    string s;
    cin >> s;
    vector<int> a(10);
    for (char c : s) { // 记录0~9出现的次数
        a[c - '0']++;
    }
    int len;
    cin >> len;
    vector<int> b(10);
    for (int i = 1, l = 1; i <= 1000; i++) { // 右指针递增
        for (char c : to_string(i)) {
            b[c - '0']++;
        }
        int tot = 0;
        for (int i = 0; i <= 9; i++) {
            while (b[i] > a[i]) { // 出现b数组中的个数大于a数组中的个数
                for (char c : to_string(l++)) { // 左指针递增
                    b[c - '0']--;
                }
            }
            tot += b[i] == a[i];
        }
        if (tot == 10 && i - l + 1 == len) {
            cout << l << endl;
            break;
        }
    }
}

import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String s = scanner.next();
        int[] a = new int[10];
        for (char c : s.toCharArray()) { // 记录0~9出现的次数
            a[c - '0']++;
        }
        int len = scanner.nextInt();
        int[] b = new int[10];
        for (int i = 1, l = 1; i <= 1000; i++) { // 右指针递增
            for (char c : String.valueOf(i).toCharArray()) {
                b[c - '0']++;
            }
            int tot = 0;
            for (int j = 0; j <= 9; j++) {
                while (b[j] > a[j]) { // 出现b数组中的个数大于a数组中的个数
                    for (char c : String.valueOf(l++).toCharArray()) { // 左指针递增
                        b[c - '0']--;
                    }
                }
                if (b[j] == a[j]) tot++;
            }
            if (tot == 10 && i - l + 1 == len) {
                System.out.println(l);
                break;
            }
        }
    }
}

const readline = require('readline');

const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
});

rl.on('line', line => {
    const [s, len] = line.split(" ");
    const a = new Array(10).fill(0);
    for (const c of s) { // 记录0~9出现的次数
        a[parseInt(c)]++;
    }
    const b = new Array(10).fill(0);
    let l = 1;
    for (let i = 1; i <= 1000; i++) { // 右指针递增
        for (const c of i.toString()) {
            b[parseInt(c)]++;
        }
        let tot = 0;
        for (let j = 0; j <= 9; j++) {
            while (b[j] > a[j]) { // 出现b数组中的个数大于a数组中的个数
                for (const c of l.toString()) { // 左指针递增
                    b[parseInt(c)]--;
                }
                l++;
            }
            tot += b[j] === a[j];
        }
        if (tot === 10 && i - l + 1 === parseInt(len)) {
            console.log(l);
            break;
        }
    }

    rl.close();
});

# Go

package main

import (
	"fmt"
	"strconv"
)

func main() {
	var s string
	fmt.Scan(&s)

	a := make([]int, 10)
	for _, c := range s { // 记录0~9出现的次数
		a[int(c-'0')]++
	}

	var len int
	fmt.Scan(&len)

	b := make([]int, 10)
	for i, l := 1, 1; i <= 1000; i++ { // 右指针递增
		for _, c := range strconv.Itoa(i) {
			b[int(c-'0')]++
		}
		tot := 0
		for i := 0; i <= 9; i++ {
			for b[i] > a[i] { // 出现b数组中的个数大于a数组中的个数
				for _, c := range strconv.Itoa(l) { // 左指针递增
					b[int(c-'0')]--
				}
				l++
			}
			if b[i] == a[i] {
				tot++
			}
		}
		if tot == 10 && i-l+1 == len {
			fmt.Println(l)
			break
		}
	}
}
