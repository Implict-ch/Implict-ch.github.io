# 题目内容

塔子哥是公司里的一名领导，他负责组织今年的团建活动。这个团建活动很重要，因为公司里的员工们都非常忙碌，很少有机会聚在一起，所以这次活动是大家期待已久的。

为了让所有人都能够参加到自己想要参加的活动，塔子哥收集了所有人的意愿，并准备了三个不同的活动：AAA、BBB 和 CCC，每个活动都有不同的人数上限和参加费用。

为了让所有人都能参加到自己想要参加的活动，塔子哥收集了所有人的投票结果。现在他希望在满足所有人的意愿的前提下，尽可能地降低团建的总费用。

但是，塔子哥面临着一个难题，因为人们的意愿各不相同，他需要考虑每个人的选择，才能做出最好的决策。

因此，他找到了你，希望你能够帮助他找到一个最佳的团建方案，以让大家都能开心参与，并且尽可能地降低费用。

# 输入描述

第一行，一个整数 NNN ，代表准备参加活动的人数。（ 1≤N≤1001\le N\le 1001≤N≤100 ）

接下来 NNN 行，每行一个由 "ABC" 组成的字符串，其中第 iii 行表示第 iii 个人投票了哪几个活动。（输入保证字符串非空，且由大写的 "ABC" 字符组成）

最后 333 行，每行两个整数，分别表示三个活动的人数上限以及每个人参加的费用。（人数上限以及参与活动的费用均为不超过 100100100 的正整数）

# 输出描述

输出共 222 行

如果能满足所有人的要求，第一行输出 "YES" ，第二行输出最少的总费用。

如果不能满足所有人的要求，第一行输出 "NO" ，第二行输出最多能满足多少人。

总共需要费用：1∗2+2∗2+3∗1=91*2+2*2+3*1=91∗2+2∗2+3∗1=9

# 思路：动态规划

# 动态规划

首先需要判断是否可以让每个人都参与活动。

f[turn][j][k]f[turn][j][k]f[turn][j][k] 表示前 turnturnturn 个人中，恰好 有 iii 个人 参与活动 AAA，jjj 个人参与活动 BBB 的条件下，参与活动 CCC 的人数。

对于所有可能的 f[n][i][j]f[n][i][j]f[n][i][j]，max{i+j+f[n][i][j]}\max\{i + j + f[n][i][j]\}max{i+j+f[n][i][j]} 就是可以参与活动的最大人数，这个人数小于 nnn 说明有人不能参与活动，这个人数大于等于 nnn 说明所有人都可以参与活动。

接下来考虑所有人都可以参与活动的情况下，最小花费是多少。

dp[turn][i][j][k]dp[turn][i][j][k]dp[turn][i][j][k] 表示前 turnturnturn 个人中，恰好 有 iii 个人参加活动 AAA，jjj 个人参与活动 BBB，kkk 个人参加活动 CCC 的最小花费。

具体状态转移细节见代码。

# CPP

#include <bits/stdc++.h>
using namespace std;

const int INF = 0x3f3f3f3f;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    string s;
    vector<bool> canA(n + 1), canB(n + 1), canC(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> s;
        for (auto c: s) {
            if (c == 'A') canA[i] = true;
            if (c == 'B') canB[i] = true;
            if (c == 'C') canC[i] = true;
        }
    }

    int cntA, weightA, cntB, weightB, cntC, weightC;
    cin >> cntA >> weightA;
    cin >> cntB >> weightB;
    cin >> cntC >> weightC;

    // f[turn][i][j] 三维DP 只需要开两维即可，是背包型DP常见优化：滚动数组优化，节省空间
    vector f(cntA + 1, vector<int>(cntB + 1, -INF));
    // 起始位置
    f[0][0] = 0;

    int max_people = 0;
    for (int turn = 1; turn <= n; ++turn) {
        for (int i = min(turn, cntA); i >= 0; --i) {
            for (int j = min(turn, cntB); j >= 0; --j) {
                // 如果第 turn 个人意愿中有活动 C ，则 f[i][j] = max(f[i][j], f[i][j] + 1)
                if (canC[turn]) f[i][j] += 1;
                // 如果第 turn 个人意愿有活动 A ，表示第 turn 个人选择了 A ，则 f[i][j] = max(f[i][j], f[i - 1][j])
                if (i > 0 && canA[turn]) f[i][j] = max(f[i][j], f[i - 1][j]);
                // 如果第 turn 个人意愿有活动 B ，表示第 turn 个人选择了 B ，则 f[i][j] = max(f[i][j], f[i][j - 1])
                if (j > 0 && canB[turn]) f[i][j] = max(f[i][j], f[i][j - 1]);

                // 当 f[i][j] >= 0，说明恰好前 turn 个人中，恰好 i 个人选择活动 A ，j 个人选择了活动 B
                // 的情况下，选择活动 C 的最大人数
                // i + j + f[i][j] 表示前 turn 个人中 i
                if (f[i][j] >= 0) {
                    max_people = max(max_people, i + j + min(f[i][j], cntC));
                }
            }
        }
    }

    // 无论怎么安排，都无法让 n 个人都参与活动
    if (max_people < n) {
        cout << "NO\n";
        cout << max_people << "\n";
        return 0;
    }

    // dp[turn][i][j][k] 四维DP 只需要开三维即可，是背包型DP常见优化：滚动数组优化，节省空间
    vector dp(cntA + 1, vector(cntB + 1, vector(cntC + 1, INF)));
    // 起始位置
    dp[0][0][0] = 0;
    for (int turn = 1; turn <= n; ++turn) {
        for (int i = min(turn, cntA); i >= 0; --i)
            for (int j = min(turn - i, cntB); j >= 0; --j) {
                int k = turn - i - j;
                // 满足前 turn 个人有 i + j + k = turn 的方案才是合法方案
                if (k > cntC) continue;

                int& res = dp[i][j][k];
                // 第 i 个人选择活动 A ，dp[i][j][k] = min(dp[i][j][k], dp[i - 1][j][k] + weightA)
                if (i > 0 && canA[turn]) res = min(res, dp[i - 1][j][k] + weightA);
                // 第 i 个人选择活动 B ，dp[i][j][k] = min(dp[i][j][k], dp[i][j - 1][k] + weightB)
                if (j > 0 && canB[turn]) res = min(res, dp[i][j - 1][k] + weightB);
                // 第 i 个人选择活动 C ，dp[i][j][k] = min(dp[i][j][k], dp[i][j][k - 1] + weightC)
                if (k > 0 && canC[turn]) res = min(res, dp[i][j][k - 1] + weightC);
            }
    }

    // 此时的 dp[i][j][k] 为 dp[n][i][j][k]，即前 n 个人中有 i 个 A，j 个 B，k 个 C 的最小花费
    int min_cost = INF;
    for (int i = 0; i <= cntA; ++i) {
        for (int j = 0; j <= cntB; ++j) {
            int k = n - i - j;
            // 找到一种合法方案
            if (k <= cntC) {
                min_cost = min(min_cost, dp[i][j][k]);
            }
        }
    }

    cout << "YES\n";
    cout << min_cost << "\n";

    return 0;
}

package main

import "fmt"

const INF = 0x3f3f3f3f

func main() {
	var n int
	var s string

	fmt.Scan(&n)

	canA, canB, canC := make([]bool, n+1), make([]bool, n+1), make([]bool, n+1)
	for i := 1; i <= n; i++ {
		fmt.Scan(&s)
		for _, c := range s {
			if c == 'A' {
				canA[i] = true
			} else if c == 'B' {
				canB[i] = true
			} else if c == 'C' {
				canC[i] = true
			}
		}
	}

	var cntA, weightA, cntB, weightB, cntC, weightC int
	fmt.Scan(&cntA, &weightA, &cntB, &weightB, &cntC, &weightC)

	// f[turn][i][j] 三维DP 只需要开两维即可，是背包型DP常见优化：滚动数组优化，节省空间
	f := make([][]int, cntA+1)
	for i := range f {
		f[i] = make([]int, cntB+1)
		for j := range f[i] {
			f[i][j] = -INF
		}
	}

	// 起始位置
	f[0][0] = 0

	maxPeople := 0
	for turn := 1; turn <= n; turn++ {
		for i := min(turn, cntA); i >= 0; i-- {
			for j := min(turn, cntB); j >= 0; j-- {
				// 如果第 turn 个人意愿中有活动 C ，则 f[i][j] = max(f[i][j], f[i][j] + 1)
				if canC[turn] {
					f[i][j] += 1
				}

				// 如果第 turn 个人意愿有活动 A ，表示第 turn 个人选择了 A ，则 f[i][j] = max(f[i][j], f[i - 1][j])
				if i > 0 && canA[turn] {
					f[i][j] = max(f[i][j], f[i-1][j])
				}

				// 如果第 turn 个人意愿有活动 B ，表示第 turn 个人选择了 B ，则 f[i][j] = max(f[i][j], f[i][j - 1])
				if j > 0 && canB[turn] {
					f[i][j] = max(f[i][j], f[i][j-1])
				}

				// 当 f[i][j] >= 0，说明恰好前 turn 个人中，恰好 i 个人选择活动 A ，j 个人选择了活动 B
				// 的情况下，选择活动 C 的最大人数
				// i + j + f[i][j] 表示前 turn 个人中 i 个人选择活动 A，j 个人选择了活动 B，同时有 f[i][j] 个人选择了活动 C
				if f[i][j] >= 0 {
					maxPeople = max(maxPeople, i+j+min(f[i][j], cntC))
				}
			}
		}
	}

	// 无论怎么安排，都无法让 n 个人都参与活动
	if maxPeople < n {
		fmt.Println("NO")
		fmt.Println(maxPeople)
		return
	}

	// dp[turn][i][j][k] 四维DP 只需要开三维即可，是背包型DP常见优化：滚动数组优化，节省空间
	dp := make([][][]int, cntA+1)
	for i := 0; i <= cntA; i++ {
		dp[i] = make([][]int, cntB+1)
		for j := 0; j <= cntB; j++ {
			dp[i][j] = make([]int, cntC+1)
			for k := 0; k <= cntC; k++ {
				dp[i][j][k] = INF
			}
		}
	}

	// 起始位置
	dp[0][0][0] = 0
	for turn := 1; turn <= n; turn++ {
		for i := min(turn, cntA); i >= 0; i-- {
			for j := min(turn-i, cntB); j >= 0; j-- {
				k := turn - i - j
				// 满足前 turn 个人有 i + j + k = turn 的方案才是合法方案
				if k > cntC {
					continue
				}

				// 第 i 个人选择活动 A ，dp[i][j][k] = min(dp[i][j][k], dp[i - 1][j][k] + weightA)
				if i > 0 && canA[turn] {
					dp[i][j][k] = min(dp[i][j][k], dp[i-1][j][k]+weightA)
				}
				// 第 i 个人选择活动 B ，dp[i][j][k] = min(dp[i][j][k], dp[i][j - 1][k] + weightB)
				if j > 0 && canB[turn] {
					dp[i][j][k] = min(dp[i][j][k], dp[i][j-1][k]+weightB)
				}
				// 第 i 个人选择活动 C ，dp[i][j][k] = min(dp[i][j][k], dp[i][j][k - 1] + weightC)
				if k > 0 && canC[turn] {
					dp[i][j][k] = min(dp[i][j][k], dp[i][j][k-1]+weightC)
				}
			}
		}
	}

	// 此时的 dp[i][j][k] 为 dp[n][i][j][k]，即前 n 个人中有 i 个 A，j 个 B，k 个 C 的最小花费
	minCost := INF
	for i := 0; i <= cntA; i++ {
		for j := 0; j <= cntB; j++ {
			k := n - i - j
			// 找到一种合法方案
			if k <= cntC {
				minCost = min(minCost, dp[i][j][k])
			}
		}
	}

	fmt.Println("YES")
	fmt.Println(minCost)
}

process.stdin.on('end', () => {
    const lines = input.trim().split('\n');
    const n = parseInt(lines[0]);
    let canA = new Array(n + 1).fill(false);
    let canB = new Array(n + 1).fill(false);
    let canC = new Array(n + 1).fill(false);
    for (let i = 1; i <= n; i++) {
        const s = lines[i];
        for (let j = 0; j < s.length; j++) {
            const c = s.charAt(j);
            if (c === 'A') canA[i] = true;
            if (c === 'B') canB[i] = true;
            if (c === 'C') canC[i] = true;
        }
    }
    const [cntA, weightA] = lines[n + 1].split(' ').map(Number);
    const [cntB, weightB] = lines[n + 2].split(' ').map(Number);
    const [cntC, weightC] = lines[n + 3].split(' ').map(Number);

    // f[turn][i][j] 三维DP 只需要开两维即可，是背包型DP常见优化：滚动数组优化，节省空间
    let f = new Array(cntA + 1).fill(null).map(() => new Array(cntB + 1).fill(-Infinity));
    // 起始位置
    f[0][0] = 0;
    let max_people = 0;
    for (let turn = 1; turn <= n; turn++) {
        for (let i = Math.min(turn, cntA); i >= 0; i--) {
            for (let j = Math.min(turn, cntB); j >= 0; j--) {
                // 如果第 turn 个人意愿中有活动 C ，则 f[i][j] = max(f[i][j], f[i][j] + 1)
                if (canC[turn]) f[i][j] += 1;
                // 如果第 turn 个人意愿有活动 A ，表示第 turn 个人选择了 A ，则 f[i][j] = max(f[i][j], f[i - 1][j])
                if (i > 0 && canA[turn]) f[i][j] = Math.max(f[i][j], f[i - 1][j]);
                // 如果第 turn 个人意愿有活动 B ，表示第 turn 个人选择了 B ，则 f[i][j] = max(f[i][j], f[i][j - 1])
                if (j > 0 && canB[turn]) f[i][j] = Math.max(f[i][j], f[i][j - 1]);

                // 当 f[i][j] >= 0，说明恰好前 turn 个人中，恰好 i 个人选择活动 A ，j 个人选择了活动 B
                // 的情况下，选择活动 C 的最大人数
                // i + j + f[i][j] 表示前 turn 个人中 i 个人选择活动 A，j 个人选择活动 B，f[i][j] 个人选择活动 C
                if (f[i][j] >= 0) {
                    max_people = Math.max(max_people, i + j + Math.min(f[i][j], cntC));
                }
            }
        }
    }

    // 无论怎么安排，都无法让 n 个人都参与活动
    if (max_people < n) {
        console.log('NO');
        console.log(max_people);
        return;
    }

    // dp[turn][i][j][k] 四维DP 只需要开三维即可，是背包型DP常见优化：滚动数组优化，节省空间
    let dp = new Array(cntA + 1).fill(null).map(() => new Array(cntB + 1).fill(null).map(() => new Array(cntC + 1).fill(Infinity)));
    // 起始位置
    dp[0][0][0] = 0;
    for (let turn = 1; turn <= n; turn++) {
        for (let i= Math.min(turn, cntA); i >= 0; i--) {
            for (let j = Math.min(turn - i, cntB); j >= 0; j--) {
                const k = turn - i - j;
                if (k > cntC) continue;
                // 如果第 turn 个人意愿有活动 A ，表示第 turn 个人选择了 A ，则 dp[i][j][k] = min(dp[i][j][k], dp[i - 1][j][k] + weightA)
                if (i > 0 && canA[turn]) dp[i][j][k] = Math.min(dp[i][j][k], dp[i - 1][j][k] + weightA);
                // 如果第 turn 个人意愿有活动 B ，表示第 turn 个人选择了 B ，则 dp[i][j][k] = min(dp[i][j][k], dp[i][j - 1][k] + weightB)
                if (j > 0 && canB[turn]) dp[i][j][k] = Math.min(dp[i][j][k], dp[i][j - 1][k] + weightB);
                // 如果第 turn 个人意愿有活动 C ，表示第 turn 个人选择了 C ，则 dp[i][j][k] = min(dp[i][j][k], dp[i][j][k - 1] + weightC)
                if (k > 0 && canC[turn]) dp[i][j][k] = Math.min(dp[i][j][k], dp[i][j][k - 1] + weightC);
            }
        }
    }

    let min_cost = Infinity;
    for (let i = 0; i <= cntA; i++) {
        for (let j = 0; j <= cntB; j++) {
            const k = n - i - j;
            if (k <= cntC) {
                // 在满足 n 人都能参加的前提下，找到花费最小的方案
                min_cost = Math.min(min_cost, dp[i][j][k]);
            }
        }
    }

    console.log('YES');
    console.log(min_cost);
});
