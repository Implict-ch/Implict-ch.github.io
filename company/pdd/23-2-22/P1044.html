# 题目内容

塔子哥生活在美丽的华光林，他非常喜欢这里的环境和氛围。在平常的休息时间，他会在家附近散步，欣赏美景，感受大自然的魅力。顺便会投喂经过遇到的小马们，让它们感受到人类的温暖和关爱。这样的生活让他感到非常满足和幸福。已知华光林共有 NNN 个广场(分别编号 111 ~ NNN ，塔子哥住在 111 号广场)，以及有 N−1N-1N−1 条道路，保证广场两两之间相互连通，且只有唯一一条通路， 且已知第 iii 个广场有 AiA_iA​i​​ 只马。

塔子哥平时散步的时候有以下这些习惯：

  1. 任意一个广场不能两次访问
  2. 塔子哥的背包只能装下最多MMM份马粮
  3. 111 份马粮只能投喂 111 只马
  4. 若要经过某个广场则要投喂其中所有的马(^. w .^)

塔子哥想知道，应当如何选择路线使得能投喂到最多数量的广场，且需要的马粮数量最少。

# 输入描述

第一行，两个整数 NNN 和 MMM ，分别表示华光林广场的数量，和准备带出门的马粮的份数( 1≤N≤50,0001\le N\le 50,0001≤N≤50,000 ， 0≤M≤5,000,0000\le M\le 5,000,0000≤M≤5,000,000 )

第二行，共 NNN 个整数，其中第 iii 个整数 AiA_iA​i​​ 表示第 iii 个广场的马的数量。( 0≤Ai≤1000\le A_i \le 1000≤A​i​​≤100 )

接下来 N−1N-1N−1 行，每行两个整数 XXX 和 YYY ，表示编号 XXX 和 YYY 的广场之间存在一条道路。( 1≤X,Y≤N1\le X,Y\le N1≤X,Y≤N )

# 输出描述

共一行，两个整数，分别表示：能投喂最多数量的广场，以及在此情况下需要最少多少马粮。

# 样例

输入

2 1
2 1
1 2

输出

0 0

样例解释

塔子哥只有 111 份马粮，小于 111 号广场的马马数量，因此塔子哥决定不出门喂马。(T V T)

没有任何一个广场的马咪被投喂，也就不需要任何马粮。

# 样例二

输入

3 4
1 2 3
1 2
2 3

输出

2 3

# 思路：搜索（BFS/DFS）

NNN 个点 N−1N-1N−1 条边，且图是连通的，所以这是一棵树。

以 111 号点为根，到达其他点的最短路径都是唯一的，故从 111 号点出发到达的最远的点的路径上的广场最多，这个最远的点必然是树上的叶子。

对于所有根到最远叶子的路径，路径上需要投喂的马粮取最小值即可。

代码使用 BFS 实现，DFS 也可以。

时间复杂度：O(n)O(n)O(n)

# CPP

#include <bits/stdc++.h>
using namespace std;

int main()
{

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<int> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    // 建图
    vector<vector<int>> g(n);
    for (int i = 1; i < n; ++i) {
        int u, v;
        cin >> u >> v;
        u -= 1;
        v -= 1;
        g[u].emplace_back(v);
        g[v].emplace_back(u);
    }

    // food[i] 表示从 0 走到 i 的马粮数
    // dist[i] 表示从 0 走到 i 的广场数
    vector<int> food(n, -1);
    vector<int> dist(n, -1);
    dist[0] = 1;
    food[0] = a[0];

    // 队列 BFS 
    queue<int> q;
    q.push(0);

    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v: g[u]) {
            // 此判断用来避免死循环
            // (u, v) 这条边被建立了两次
            // 如果从 u 到了 v，那么如果没有该判断，v 再到 u，如此形成死循环
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                food[v] = food[u] + a[v];
                q.push(v);
            }
        }
    }

    // 虽然题解中说明只取叶子的值，但是非叶子点的广场数必然不是最大
    // 故这必然不可能成为答案，枚举根到所有点即可
    int ansd = 0, ansf = 0;
    for (int i = 0; i < n; ++i) {
        if (dist[i] > ansd && food[i] <= m) {
            ansd = dist[i];
            ansf = food[i];
        } else if (dist[i] == ansd && food[i] < ansf) {
            ansf = food[i];
        }
    }

    cout << ansd << " " << ansf << "\n";

    return 0;
}

        StringTokenizer st = new StringTokenizer(br.readLine());

        int n = Integer.parseInt(st.nextToken());

        st = new StringTokenizer(br.readLine());
        for (int i = 0; i < n; ++i) {
            a[i] = Integer.parseInt(st.nextToken());
        }

        // 建图
        List<List<Integer>> g = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            g.add(new ArrayList<>());
        }
        for (int i = 1; i < n; ++i) {
            st = new StringTokenizer(br.readLine());
            int u = Integer.parseInt(st.nextToken()) - 1;
            int v = Integer.parseInt(st.nextToken()) - 1;
            g.get(u).add(v);
            g.get(v).add(u);
        }

        // food[i] 表示从 0 走到 i 的马粮数
        // dist[i] 表示从 0 走到 i 的广场数
        int[] food = new int[n];
        int[] dist = new int[n];
        Arrays.fill(food, -1);
        Arrays.fill(dist, -1);
        dist[0] = 1;
        food[0] = a[0];

        // 队列 BFS
        Queue<Integer> q = new LinkedList<>();
        q.offer(0);

        while (!q.isEmpty()) {
            int u = q.poll();
            for (int v : g.get(u)) {
                // 此判断用来避免死循环
                // (u, v) 这条边被建立了两次
                // 如果从 u 到了 v，那么如果没有该判断，v 再到 u，如此形成死循环
                if (dist[v] == -1) {
                    dist[v] = dist[u] + 1;
                    food[v] = food[u] + a[v];
                    q.offer(v);
                }
            }
        }

        // 虽然题解中说明只取叶子的值，但是非叶子点的广场数必然不是最大
        // 故这必然不可能成为答案，枚举根到所有点即可

        for (int i = 0; i < n; ++i) {
            if (dist[i] > ansd && food[i] <= m) {
                ansd = dist[i];
                ansf = food[i];
            } else if (dist[i] == ansd && food[i] < ansf) {
                ansf = food[i];
            }
        }

package main

import "fmt"

func main() {
    var n, m int
    fmt.Scan(&n, &m)

    a := make([]int, n)
    for i := 0; i < n; i++ {
        fmt.Scan(&a[i])
    }

    // 建图
    g := make([][]int, n)
    for i := 1; i < n; i++ {
        var u, v int
        fmt.Scan(&u, &v)
        u -= 1
        v -= 1
        g[u] = append(g[u], v)
        g[v] = append(g[v], u)
    }

    // food[i] 表示从 0 走到 i 的马粮数
    // dist[i] 表示从 0 走到 i 的广场数
    food := make([]int, n)
    dist := make([]int, n)
    for i := 0; i < n; i++ {
        food[i] = -1
        dist[i] = -1
    }
    dist[0] = 1
    food[0] = a[0]

    // 队列 BFS
    q := make([]int, 0)
    q = append(q, 0)

    for len(q) > 0 {
        u := q[0]
        q = q[1:]
        for _, v := range g[u] {
            // 此判断用来避免死循环
            // (u, v) 这条边被建立了两次
            // 如果从 u 到了 v，那么如果没有该判断，v 再到 u，如此形成死循环
            if dist[v] == -1 {
                dist[v] = dist[u] + 1
                food[v] = food[u] + a[v]
                q = append(q, v)
            }
        }
    }

    // 虽然题解中说明只取叶子的值，但是非叶子点的广场数必然不是最大
    // 故这必然不可能成为答案，枚举根到所有点即可
    ansd, ansf := 0, 0
    for i := 0; i < n; i++ {
        if dist[i] > ansd && food[i] <= m {
            ansd = dist[i]
            ansf = food[i]
        } else if dist[i] == ansd && food[i] < ansf {
            ansf = food[i]
        }
    }

    fmt.Printf("%d %d\n", ansd, ansf)
}
