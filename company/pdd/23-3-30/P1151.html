# 题目内容

塔子哥是一位资深的投资人，他经常在股市中寻找投资机会。

最近，他得到了一个长度为 nnn 的数组，数组中的元素均为 [−8,−4,−2,−1,0,1,2,4,8][-8,-4,-2,-1,0,1,2,4,8][−8,−4,−2,−1,0,1,2,4,8] 中的任意一个。

他发现这个数组中存在一些连续的子数组，它们的乘积可能会很大。他希望能够找到乘积最大的连续子数组，并输出它的左右端点。

# 输入描述

输入第一行为一个整数 nnn ，（ 1≤n≤1051\le n \le 10^51≤n≤10​5​​ ）

第二行为 nnn 个整数，每个整数均为 [−8,−4,−2,−1,0,1,2,4,8][-8,-4,-2,-1,0,1,2,4,8][−8,−4,−2,−1,0,1,2,4,8] 中的任意一个。

# 输出描述

输出乘积最大的连续子数组的左右端点。如果有多个符合要求的子数组，输出左端点最小的那个方案。如果仍然有多个，输出右端点最小的那个方案。

# 样例

输入

5
-1 4 -2 -4 4

输出

2 5

# 样例2

输入

3
-1 0 -2

输出

1 2

# 思路：分类讨论 + 贪心

当有正数，或者至少两个负数，则答案必然是正数。

首先考虑答案值不为正数的情况，即只有 000和至多 111 个负数的情况。

如果只有一个负数，那么答案就是这个负数。

如果答案中存在000，那么答案区间必然是包括 000 的。假设最靠近111 的 000 的索引为 iii。 答案要求左端点尽可能小，则左端点设置为 111，右端点设置为$ i。如此答案为 0$，答案区间也是最优444的。

接下来考虑答案值为正数的情况。

首先，因为累乘结果很大，231−12^{31}-12​31​​−1 是 32位 int 最大值，263−12^{63}-12​63​​−1 是 646464位 $int $最大值。故不能直接乘。因为只会出现 {−8,−4,−2,−1,0,1,2,4,8}\{-8, -4, -2, -1, 0, 1, 2, 4, 8\}{−8,−4,−2,−1,0,1,2,4,8} 这九种数。都是 222 的幂，故可以将乘法转换为加法。

这里因为答案必然是正数，而如果区间中存在$ 0，整个区间之积必然为 0$，故以 $0 $作为分界划分为多个部分。

接下来我们只考虑存在正数和负数的一个数组，原问题就是多个数组的最大值。

对于这个数组来说，考虑前 iii 个数，dp[i]dp[i]dp[i] 表示包括第 iii 个数的乘积最大值：

  • 如果前 iii 个数中负数的个数为偶数，则这 iii 个数相乘的值必然是最大值。但是需要注意一点，如果乘的是 111 或者连续两个 −1-1−1，则值不变，即我们的左端点为 111 ，但是右端点更大了，此时不能更新答案。

  • 如果前 iii 个数中负数的个数为奇数，找到从左到右第一个负数 aja_ja​j​​，aj+1×aj+2×⋯×aia_{j+1}\times a_{j+2}\times \cdots\times a_ia​j+1​​×a​j+2​​×⋯×a​i​​， 必然是正数，然后根据这些数乘积大小来更新最大值和对应的区间。

时间复杂度：O(n)O(n)O(n)

# CPP

#include <bits/stdc++.h>
using namespace std;

typedef pair<int, int> PII;

int get2(int x) {
    int bit = 0;
    while ((1 << bit) < x) bit += 1;
    return bit;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    int maxv = -10, l = -1, r = -1;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        // 这里主要是为了处理都是 0 和至多一个负数的情况
        if (a[i] > maxv) {
            maxv = a[i];
            // 如果是负数，就是其自己
            if (a[i] != 0) l = i + 1, r = i + 1;
            // 如果是 0，则左端点必须是 1
            else l = 1, r = i + 1;
        }
    }

    // 根据 0 为分界线，分成若干组，每组只有正数和负数
    vector<vector<PII>> vec;
    for (int i = 0; i < n; ++i) {
        if (a[i] == 0) continue;
        vector<PII> temp;
        int j = i;
        while (j < n && a[j] != 0) {
            // 以 0 为分界线，将所有连续的，不为 0 的数分为一组
            temp.emplace_back(a[j], j + 1);
            j += 1;
        }
        vec.emplace_back(temp);
        i = j - 1;
    }

    // 这里初始化为 -1 是因为，1 的幂为 0，如果初始化为 0，则不会更答案，但是 1 确实是正数
    int max_num_val = -1;
    for (auto& u: vec) {
        // 选偶数个负数的组
        // 当负数的个数为偶数，就是统计前缀
        // 当负数的个数为奇数（奇数>=3），就是统计第一个奇数后面的数的和

        // 第一个负数对应的索引
        int first_neg_idx = -1;
        // 第一个负数及其之前的值的幂和
        int before_first_neg_val = 0;
        // 前 i 个数幂和
        int all_num_val = 0;
        // 前 i 个数中负数个数
        int neg_cnt = 0;

        for (int i = 0; i < u.size(); ++i) {
            // 更新前 i 个数的幂和
            all_num_val += get2(abs(u[i].first));
            if (u[i].first < 0) {
                neg_cnt += 1;
                // 如果是第一个负数
                if (first_neg_idx == -1) {
                    // 记录每组第一个负数的索引
                    first_neg_idx = u[i].second;
                    // 计算第一个负数及其之前的数的幂和
                    before_first_neg_val = all_num_val;
                    // 如果是第一个负数，则其和之前的数相乘的结果必然是负数，此时我们的答案必然大于 0，没必要更新答案
                    continue;
                }
            }

            // 如果前 i 个数中负数的个数为偶数，则最大值必然是前 i 个数之积
            if (neg_cnt % 2 == 0) {
                // 只有值更大，才会更新，否则会将 l 或者 r 变得更大，就不再是最优答案了
                if (all_num_val > max_num_val) {
                    max_num_val = all_num_val;
                    l = u[0].second;
                    r = u[i].second;
                }
            } else {
                // 如果前 i 个数中负数的个数为奇数，则计算从第一个负数之后到当前数的之积，如果更大则更新答案
                if (all_num_val - before_first_neg_val > max_num_val) {
                    max_num_val = all_num_val - before_first_neg_val;
                    l = first_neg_idx + 1;
                    r = u[i].second;
                }
            }
        }
    }

    cout << l << " " << r << "\n";

    return 0;
}

package main

import (
	"bufio"
	"fmt"
	"os"
)

type PII struct {
	first, second int
}

func abs(x int) int {
	if x < 0 {
		return -x
	}
	return x
}

func get2(x int) int {
	bit := 0
	for 1<<bit < x {
		bit++
	}
	return bit
}

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)

	var n int
	fmt.Scan(&n)

	a := make([]int, n)
	for i := 0; i < n; i++ {
		fmt.Scan(&a[i])
	}

	maxv, l, r := -10, -1, -1

	// 找到序列中最大的数及其下标，如果最大的数不是 0，那么左右端点都是它自己，如果最大的数是 0，则左端点必须是 1
	for i := 0; i < n; i++ {
		if a[i] > maxv {
			maxv = a[i]
			if a[i] != 0 {
				l, r = i+1, i+1
			} else {
				l, r = 1, i+1
			}
		}
	}

	vec := make([][]PII, 0)

	// 根据 0 为分界线，将序列分成若干组，每组只有正数和负数
	for i := 0; i < n; i++ {
		if a[i] == 0 {
			continue
		}
		temp := make([]PII, 0)
		j := i
		for j < n && a[j] != 0 {
			temp = append(temp, PII{a[j], j + 1})
			j++
		}
		vec = append(vec, temp)
		i = j - 1
	}

	max_num_val := -1

	for _, u := range vec {
		// 选偶数个负数的组
		// 当负数的个数为偶数，就是统计前缀幂和

		before_first_neg_val := 0
		// 前 i 个数幂和
		all_num_val := 0
		// 前 i 个数中负数个数
		neg_cnt := 0

		for i := 0; i < len(u); i++ {
			// 更新前 i 个数的幂和

				// 如果是第一个负数

					// 记录每组第一个负数的索引
					first_neg_idx = u[i].second
					// 计算第一个负数及其之前的数的幂和
					before_first_neg_val = all_num_val
					// 如果是第一个负数，则其和之前的数相乘的结果必然是负数，此时我们的答案必然大于 0，没必要更新答案
					continue
				}
			}

			if neg_cnt%2 == 0 {
				// 只有值更大，才会更新，否则会将 l 或者 r 变得更大，就不再是最优答案了
				if all_num_val > max_num_val {

				// 如果是第一个负数
				if (first_neg_idx === -1) {
					// 记录每组第一个负数的索引
					first_neg_idx = u[i][1];
					// 计算第一个负数及其之前的数的幂和
					before_first_neg_val = all_num_val;
					// 如果是第一个负数，则其和之前的数相乘的结果必然是负数，此时我们的答案必然大于 0，没必要更新答案
					continue;
				}
			}

			// 如果前 i 个数中负数的个数为偶数，则最大值必然是前 i 个数之积
			if (neg_cnt % 2 === 0) {
				// 只有值更大，才会更新，否则会将 l 或者 r 变得更大，就不再是最优答案了
				if (all_num_val > max_num_val) {
					max_num_val = all_num_val;
					l = u[0][1];
					r = u[i][1];
				}
			} else {
				// 如果前 i 个数中负数的个数为奇数，则计算从第一个负数之后到当前数的之接上一段代码：

				//积，如果更大则更新答案
				if (all_num_val - before_first_neg_val > max_num_val) {
					max_num_val = all_num_val - before_first_neg_val;
					l = first_neg_idx + 1;
					r = u[i][1];
				}
			}
		}
	}

	console.log(l + " " + r);
});
