# 题目内容

塔子哥是一名拥有独特天赋的冒险者，他拥有一双敏锐的眼睛，能够寻找到别人无法察觉的宝石和宝藏。为了寻找更多的宝藏和宝石，他踏上了一段新的旅程。

在这段旅程中，他来到了一个神秘的游戏世界。这个游戏世界有 nnn 个关卡和 mmm 种宝石。每一个关卡都有一个bossbossboss或商店。当他打败一个bossbossboss时，他可以获得一个新的宝石。但是，由于他只能携带一个宝石，他必须丢掉当前的宝石才能取走新的宝石。此外，当他到达一个商店时，商店会收购一种特定的宝石，并给他一些钱。

塔子哥想知道自己最多可以获得多少钱。因此，他开始了他的游戏之旅，希望能在这个游戏世界中获得尽可能多的财富。

# 输入描述

输入第一行为两个整数 nnn 和 mmm ，（ 1≤n,m≤1061\le n,m\le 10^61≤n,m≤10​6​​ ）

接下来 nnn 行，每一行第一个输入为一个字符，如果字符为 b，则代表bossbossboss，后面输入一个整数，代表宝石种类。

如果字符为 m ，则代表商店，后面输入两个整数，代表该商店收取的宝石种类和价格。

# 输出描述

输出塔子哥最多可以获得多少钱。

# 样例

输入

5 3
b 3
b 1
m 3 50
b 2
m 1 100

输出

100

# 思路：DP+哈希表

dp[i]dp[i]dp[i] 表示前 iii 关，可以获得的最大财富。

pre[x]pre[x]pre[x] 表示宝石 xxx 的最新位置，由于数据范围是 10610^610​6​​，用一个数组代替哈希表即可。

dp[i]dp[i]dp[i] 是单调递增的（非严格）

  • 当第 iii 关是 boss 关，这一关不能使得财富值变大，而是更新宝石 xxx 的位置

  • 当第 iii 关是 商店关，在这一关售卖宝石 xxx 获得的财富值为 yyy，但是有一个条件是：

    这个宝石 xxx 必须在前 i−1i-1i−1 关中，最靠近当前商店关的一个 boss 关 jjj 中取走的，即 dp[i]=dp[j]+ydp[i] = dp[j] + ydp[i]=dp[j]+y

    假设我们取的 xxx 是从 boss 关 kkk 中取走的，k<j<ik < j < ik<j<i，因为 dp[i]dp[i]dp[i] 是单调递增的，dp[k]≤dp[j]dp[k]\leq dp[j]dp[k]≤dp[j]，故必然是从最靠近当前商店关的一个 boss 关 jjj 中取走 xxx 的。

# CPP

#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    // pre[x] 表示前一个 x 的位置，如果为 0 表示未出现过
    vector<int> pre(m + 1);

    // dp[i] 表示前 i 关获得的最多财富
    // dp[i] 是单调递增的（非严格）
    vector<long long> dp(n + 1);

    string op;
    int x, y;
    for (int i = 1; i <= n; ++i) {
        dp[i] = dp[i - 1];

        cin >> op >> x;
        if (op == "m") {
            // 如果是本关是商店，则考虑售卖
            // 当前关售卖宝石 x 的条件是，之前有 boss 关出现过 x
            cin >> y;
            if (pre[x]) {
                dp[i] = max(dp[i], dp[pre[x]] + y);
            }
        } else {
            // 如果本关是 boss 关，则更新宝石 x 的最新位置
            pre[x] = i;
        }
    }

    // 因为 dp[i] 是单调递增的，所以 dp[n] 一定是最大值
    cout << dp[n] << "\n";

    return 0;
}

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int n = scanner.nextInt();
        int m = scanner.nextInt();

        // pre[x] 表示前一个 x 的位置，如果为 0 表示未出现过
        int[] pre = new int[m + 1];

        // dp[i] 表示前 i 关获得的最多财富
        // dp[i] 是单调递增的（非严格）
        long[] dp = new long[n + 1];

        for (int i = 1; i <= n; ++i) {
            dp[i] = dp[i - 1];

            String op = scanner.next();
            int x = scanner.nextInt();
            if (op.equals("m")) {
                // 如果是本关是商店，则考虑售卖
                // 当前关售卖宝石 x 的条件是，之前有 boss 关出现过 x
                int y = scanner.nextInt();
                if (pre[x] != 0) {
                    dp[i] = Math.max(dp[i], dp[pre[x]] + y);
                }
            } else {
                // 如果本关是 boss 关，则更新宝石 x 的最新位置
                pre[x] = i;
            }
        }

        // 因为 dp[i] 是单调递增的，所以 dp[n] 一定是最大值
        System.out.println(dp[n]);
    }
}

# Go

package main

import (
    "bufio"
    "fmt"
    "os"
)

func main() {
    var n, m int
    fmt.Scan(&n, &m)

    // pre[x] 表示前一个 x 的位置，如果为 0 表示未出现过
    pre := make([]int, m+1)

    // dp[i] 表示前 i 关获得的最多财富
    // dp[i] 是单调递增的（非严格）
    dp := make([]int64, n+1)

    scanner := bufio.NewScanner(os.Stdin)
    scanner.Split(bufio.ScanWords)
    for i := 1; i <= n; i++ {
        dp[i] = dp[i-1]

        scanner.Scan()
        op := scanner.Text()
        scanner.Scan()
        var x, y int
        fmt.Sscan(scanner.Text(), &x)

        if op == "m" {
            // 如果是本关是商店，则考虑售卖
            // 当前关售卖宝石 x 的条件是，之前有 boss 关出现过 x
            scanner.Scan()
            fmt.Sscan(scanner.Text(), &y)
            if pre[x] != 0 {
                dp[i] = max(dp[i], dp[pre[x]]+int64(y))
            }
        } else {
            // 如果本关是 boss 关，则更新宝石 x 的最新位置
            pre[x] = i
        }
    }

    // 因为 dp[i] 是单调递增的，所以 dp[n] 一定是最大值
    fmt.Println(dp[n])
}

func max(a, b int64) int64 {
    if a > b {
        return a
    }
    return b
}
