# 题目内容

塔子哥是一名拥有独特天赋的冒险者，他拥有一双敏锐的眼睛，能够寻找到别人无法察觉的宝石和宝藏。为了寻找更多的宝藏和宝石，他踏上了一段新的旅程。

在这段旅程中，他来到了一个神秘的游戏世界。这个游戏世界有 nnn 个关卡和 mmm 种宝石。每一个关卡都有一个boss或商店。当他打败一个boss时，他可以获得一个新的宝石。但是，由于他只能携带一个宝石，他必须丢掉当前的宝石才能取走新的宝石。此外，当他到达一个商店时，商店会收购一种特定的宝石，并给他一些钱。

塔子哥想知道自己最多可以获得多少钱。因此，他开始了他的游戏之旅，希望能在这个游戏世界中获得尽可能多的财富。

# 输入描述

输入第一行为两个整数 nnn 和 mmm ，（ 1≤n,m≤1061\le n,m\le 10^61≤n,m≤10​6​​ ）

接下来 nnn 行，每一行第一个输入为一个字符，如果字符为 b，则代表boss，后面输入一个整数，代表宝石种类。

如果字符为 m ，则代表商店，后面输入两个整数，代表该商店收取的宝石种类和价格。

# 输出描述

输出塔子哥最多可以获得多少钱。

# 样例

输入

5 3
b 3
b 1
m 3 50
b 2
m 1 100

输出

100

# 思路：DFS+树形DP

本题的 DFS 类似于记忆化搜索的思想，但是由于这是一棵树，每个状态只会被计算一次，不会进入到重复的状态，但是思想是这么个思想，需要先计算 u 依赖的子节点 v 的值，然后根据子节点的值来决定当前点的值。记忆化搜索就是一种自顶向下思考，反过来的自底向上就是拓扑排序的思想。这种树上问题，当前的状态由其儿子状态转移而来的，就是一种树形dp问题。

dfs(u) 表示从 u 出发到达所有叶子，需要的最少维修队 对于 u 的每个儿子 v 需要考虑从 v 出发到达所有叶子，需要的最少维修队，这是一个原问题的子问题 求解完 dfs(v) 后，有四种情况需要考虑，其中 w(u, v) 表示 u-v 这条边是否是断边。

  • w(u, v) = 0, dfs(v) = 0，此时从 u 到 v 这棵子树的所有点均不需要任何维修队
  • w(u, v) = 1, dfs(v) = 0，此时从 u 到 v 这个点需要一个维修队，到达 v 的子树中其余点均不需要维修队
  • w(u, v) = 0, dfs(v) > 0，此时从 v 到 v 的子树中的所有点需要 dfs(v) 个维修队，则从 u 到 v 需要 dfs(v) 个维修队
  • w(u, v) = 1, dfs(v) > 0，此时从 v 到 v 的子树中的所有点需要 max(1, dfs(v)) 个维修队，因为 dfs(v) > 0, dfs(v) >= w(u, v) 是必然的，则从 u 到 v 需要 dfs(v) 个维修队

综上，dfs(u)=∑vissonofumax(dfs(v),w(u,v))dfs(u) = \sum\limits_{v\ is\ son\ of \ u} max(dfs(v), w(u,v))dfs(u)=​vissonofu​∑​​max(dfs(v),w(u,v))

时间复杂度：O(n)O(n)O(n)

# 代码

# C++

#include <bits/stdc++.h>
using namespace std;

typedef pair<int, int> PII;

// dfs(u) 表示从 u 出发到达所有叶子，需要的最少维修队
// 对于 u 的每个儿子 v
// 需要考虑从 v 出发到达所有叶子，需要的最少维修队，这是一个原问题的子问题
// 求解完 dfs(v) 后，有四种情况需要考虑
// w(u, v) = 0, dfs(v) = 0
// w(u, v) = 1, dfs(v) = 0
// w(u, v) = 0, dfs(v) >= 1
// w(u, v) = 1, dfs(v) >= 1
int dfs(int u, int fa, vector<vector<PII>>& g) {
    int res = 0;
    for (auto& [v, w]: g[u]) {
        if (v == fa) continue;
        res += max(dfs(v, u, g), w);
    }
    return res;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<vector<PII>> g(n);
    for (int i = 1; i < n; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        // 树是无向图，需要双向建图
        u--; v--;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }

    cout << dfs(0, -1, g) << "\n";

    return 0;
}

# Java

import java.util.*;
public class Main {
    
    static List<List<int[]>> g;
    
    // dfs(u) 表示从 u 出发到达所有叶子，需要的最少维修队
    // 对于 u 的每个儿子 v
    // 需要考虑从 v 出发到达所有叶子，需要的最少维修队，这是一个原问题的子问题
    // 求解完 dfs(v) 后，有四种情况需要考虑
    // w(u, v) = 0, dfs(v) = 0
    // w(u, v) = 1, dfs(v) = 0
    // w(u, v) = 0, dfs(v) >= 1
    // w(u, v) = 1, dfs(v) >= 1
    static int dfs(int u, int fa) {
        int res = 0;
        for (int[] t: g.get(u)) {
            int v = t[0], w = t[1];
            if (v == fa) continue;
            res += Math.max(w, dfs(v, u));
        }
        return res;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        g = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            g.add(new ArrayList<>());
        }

        for (int i = 1; i < n; ++i) {
            int u = sc.nextInt();
            int v = sc.nextInt();
            int w = sc.nextInt();
            u--;
            v--;
            // 树是无向图，需要双向建图
            g.get(u).add(new int[]{v, w});
            g.get(v).add(new int[]{u, w});
        }

        System.out.println(dfs(0, -1));
    }
}

# Go

package main

import "fmt"

type PII struct {
	first, second int
}

var g [][]PII

// dfs(u) 表示从 u 出发到达所有叶子，需要的最少维修队
// 对于 u 的每个儿子 v
// 需要考虑从 v 出发到达所有叶子，需要的最少维修队，这是一个原问题的子问题
// 求解完 dfs(v) 后，有四种情况需要考虑
// w(u, v) = 0, dfs(v) = 0
// w(u, v) = 1, dfs(v) = 0
// w(u, v) = 0, dfs(v) >= 1
// w(u, v) = 1, dfs(v) >= 1
func dfs(u, fa int) int {
	res := 0
	for _, p := range g[u] {
		v, w := p.first, p.second
		if v == fa {
			continue
		}

		res += max(dfs(v, u), w)
	}
	return res
}

func main() {
	var n int
	fmt.Scan(&n)
	g = make([][]PII, n)
	for i := 0; i < n; i++ {
		g[i] = make([]PII, 0)
	}

	var u, v, w int

	for i := 1; i < n; i++ {
		fmt.Scan(&u, &v, &w)
		u--
		v--

		// 树是无向图，需要双向建图
		g[u] = append(g[u], PII{v, w})
		g[v] = append(g[v], PII{u, w})
	}

	fmt.Println(dfs(0, -1))
}

func max(x, y int) int {
	if x < y {
		return y
	}
	return x
}
