# 题目内容

在这座树形城市中，有 nnn 个节点互相连接，每个节点可以作为一座城市的代表。塔子哥是这座城市的管理员，他一直关注着这座城市的安全和交通状况。

但是最近，城市中的一些道路因为年久失修、自然灾害等原因出现了断裂，这导致了交通的瘫痪和城市的分裂。为了解决这个问题，塔子哥决定组织一支维修队来修复这些断裂的道路。

维修队会从城市中的出发点 111 出发，一个维修队从 111 出发到一个点，路上所有断边都会被修好。

塔子哥想知道最少需要多少个维修队，才能修复好整座城市的断边。

# 输入描述

输入第一行为一个整数 nnn ，（ 1≤n≤1051\le n \le 10^51≤n≤10​5​​ ）

接下来 n−1n - 1n−1 行，每一行有三个整数 uuu，vvv，zzz。1≤u,v≤n1\leq u,v\leq n1≤u,v≤n， z=1z=1z=1 代表 uuu 到 vvv 的边为断的， z=0z=0z=0 代表 uuu 到 vvv 的边是好的。

# 输出描述

输出最少需要多少个修理队，能修复好整座城市的断边。

# 思路：DFS+树形DP

# CPP

#include <bits/stdc++.h>
using namespace std;

typedef pair<int, int> PII;

// dfs(u) 表示从 u 出发到达所有叶子，需要的最少维修队
// 对于 u 的每个儿子 v
// 需要考虑从 v 出发到达所有叶子，需要的最少维修队，这是一个原问题的子问题
// 求解完 dfs(v) 后，有四种情况需要考虑
// w(u, v) = 0, dfs(v) = 0
// w(u, v) = 1, dfs(v) = 0
// w(u, v) = 0, dfs(v) >= 1
// w(u, v) = 1, dfs(v) >= 1
int dfs(int u, int fa, vector<vector<PII>>& g) {
    int res = 0;
    for (auto& [v, w]: g[u]) {
        if (v == fa) continue;
        res += max(dfs(v, u, g), w);
    }
    return res;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<vector<PII>> g(n);
    for (int i = 1; i < n; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        // 树是无向图，需要双向建图
        u--; v--;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }

    cout << dfs(0, -1, g) << "\n";

    return 0;
}

import java.util.*;
public class Main {
    
    static List<List<int[]>> g;
    
    // dfs(u) 表示从 u 出发到达所有叶子，需要的最少维修队
    // 对于 u 的每个儿子 v
    // 需要考虑从 v 出发到达所有叶子，需要的最少维修队，这是一个原问题的子问题
    // 求解完 dfs(v) 后，有四种情况需要考虑
    // w(u, v) = 0, dfs(v) = 0
    // w(u, v) = 1, dfs(v) = 0
    // w(u, v) = 0, dfs(v) >= 1
    // w(u, v) = 1, dfs(v) >= 1
    static int dfs(int u, int fa) {
        int res = 0;
        for (int[] t: g.get(u)) {
            int v = t[0], w = t[1];
            if (v == fa) continue;
            res += Math.max(w, dfs(v, u));
        }
        return res;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        g = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            g.add(new ArrayList<>());
        }

        for (int i = 1; i < n; ++i) {
            int u = sc.nextInt();
            int v = sc.nextInt();
            int w = sc.nextInt();
            u--;
            v--;
            // 树是无向图，需要双向建图
            g.get(u).add(new int[]{v, w});
            g.get(v).add(new int[]{u, w});
        }

        System.out.println(dfs(0, -1));
    }
}

# Go

package main

import "fmt"

type PII struct {
	first, second int
}

var g [][]PII

// dfs(u) 表示从 u 出发到达所有叶子，需要的最少维修队
// 对于 u 的每个儿子 v
// 需要考虑从 v 出发到达所有叶子，需要的最少维修队，这是一个原问题的子问题
// 求解完 dfs(v) 后，有四种情况需要考虑
// w(u, v) = 0, dfs(v) = 0
// w(u, v) = 1, dfs(v) = 0
// w(u, v) = 0, dfs(v) >= 1
// w(u, v) = 1, dfs(v) >= 1
func dfs(u, fa int) int {
	res := 0
	for _, p := range g[u] {
		v, w := p.first, p.second
		if v == fa {
			continue
		}

		res += max(dfs(v, u), w)
	}
	return res
}

func main() {
	var n int
	fmt.Scan(&n)
	g = make([][]PII, n)
	for i := 0; i < n; i++ {
		g[i] = make([]PII, 0)
	}

	var u, v, w int

	for i := 1; i < n; i++ {
		fmt.Scan(&u, &v, &w)
		u--
		v--

		// 树是无向图，需要双向建图
		g[u] = append(g[u], PII{v, w})
		g[v] = append(g[v], PII{u, w})
	}

	fmt.Println(dfs(0, -1))
}

func max(x, y int) int {
	if x < y {
		return y
	}
	return x
}
