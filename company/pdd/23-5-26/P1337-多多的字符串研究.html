# 题目内容

塔子哥最近痴迷于研究字符串。这天塔子哥收到一封来自外太空的邮件，塔子哥打开邮件只发现了一个字符串，于是塔子哥顿时来了兴趣，他想研究一下这个字符串的含义，从中发现它的规律，解读出外星邮件的秘密。对于一个字符串sss，塔子哥想知道它所有不同的子串中，第kkk小的是什么。

# 输入描述

输入第一行包含一个字符串sss。

输入第二行包含一个整数KKK。

sss的长度不超过100100100，1<=k<=100001<=k<=100001<=k<=10000.

# 输出描述

输出第kkk小的字串，如果第kkk小的子串不存在，输出NO ANSWER

# 示例1

输入

aab
4

输出

ab

说明

字符串aab的子串从小到大依次为
a
aa
aab
ab
b

# 示例2

输入

ababababab
8

输出

abababab

# 思路：去重 + 优先队列

对于一个长度为nnn的字符串，有m=n×(n+1)2m=\frac{n\times (n+1)}{2}m=​2​​n×(n+1)​​个子字符串。

一般来说，本题求第kkk小，则自然会想到使用优先队列来维护。

但是本题要求本质不同的第kkk个子字符串，还需要去重。

所以将这mmm个子字符串存储下来，然后排序后的第kkk个字符串即为答案。

当然，本题是可以用优先队列维护的，只要保证我们对于每个相同的子字符串，只处理一次即可，用hash表即可。

# C++

// 排序去重
#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string s;
    int k;
    cin >> s >> k;

    vector<string> vec;
    // 将所有子串拿出来
    for (int i = 0; i < s.size(); ++i) {
        for (int j = i; j < s.size(); ++j) {
            vec.emplace_back(s.substr(i, j - i + 1));
        }
    }

    // 排序
    sort(vec.begin(), vec.end());
    // 去重
    vec.resize(unique(vec.begin(), vec.end()) - vec.begin());
    // 如果不足 k 个，则不存在答案
    if (k > vec.size()) cout << "NO ANSWER\n";
    // 否则索引 k-1 对应的字符串就是答案
    else cout << vec[k - 1] << "\n";

    return 0;
}

class Substring:
    def __init__(self, value):
        self.value = value
        
    def __lt__(self, other):
        return self.value > other.value

            continue
        S.add(temp)

        if len(q) < k:

            heapq.heappop(q)

# Java

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String s = sc.next();
        int k = sc.nextInt();
        int n = s.length();

        // 用于保证每个子字符串只出现一次
        Set<String> S = new HashSet<>();
        // 用于维护前 k 小子字符串，这是一个根据字典序比较的大根堆
        PriorityQueue<String> q = new PriorityQueue<>((a, b) -> b.compareTo(a));

        for (int i = 0; i < n; ++i) {
            String temp = "";
            for (int j = i; j < n; ++j) {
                temp += s.charAt(j);
                // 如果出现过，则不再考虑
                if (S.contains(temp)) continue;
                S.add(temp);

                // 如果不满 k 个，则直接加入
                if (q.size() < k) {
                    q.offer(temp);
                } else if (!q.isEmpty() && temp.compareTo(q.peek()) < 0) {
                    // 如果为 k 个，则考虑是否可以 pop 堆顶，push temp
                    q.poll();
                    q.offer(temp);
                }
            }
        }

        // 如果不足 k 个，则不存在答案
        if (q.size() < k) {
            System.out.println("NO ANSWER");
        } else {
            // 否则堆顶就是答案，第 k 小
            String kth_smallest = q.peek();
            System.out.println(kth_smallest);
        }
    }
}

package main

import (
    "container/heap"
    "fmt"
)

type PriorityQueue []string

func (pq PriorityQueue) Len() int {
    return len(pq)
}

func (pq PriorityQueue) Less(i, j int) bool {
    return pq[i] > pq[j]
}

func (pq PriorityQueue) Swap(i, j int) {
    pq[i], pq[j] = pq[j], pq[i]
}

func (pq *PriorityQueue) Push(x interface{}) {
    item := x.(string)
    *pq = append(*pq, item)
}

func (pq *PriorityQueue) Pop() interface{} {
    old := *pq
    n := len(old)
    item := old[n-1]
    *pq = old[0 : n-1]
    return item
}

func main() {
    var s string
    var k int
    fmt.Scan(&s, &k)
    n := len(s)

    // 用于保证每个子字符串只出现一次
    S := make(map[string]bool)
    // 用于维护前 k 小子字符串，这是一个根据字典序比较的大根堆
    q := make(PriorityQueue, 0)

    for i := 0; i < n; i++ {
        temp := ""
        for j := i; j < n; j++ {
            temp += string(s[j])
            // 如果出现过，则不再考虑
            if _, ok := S[temp]; ok {
                continue
            }
            S[temp] = true

            // 如果不满 k 个，则直接加入
            if len(q) < k {
                heap.Push(&q, temp)
            } else if temp < q[0] {
                // 如果为 k 个，则考虑是否可以 pop 堆顶，push temp
                heap.Pop(&q)
                heap.Push(&q, temp)
            }
        }

    // 如果不足 k 个，则不存在答案

        fmt.Println("NO ANSWER")

        kth_smallest := q[0]
        fmt.Println(kth_smallest)
    }
}

class PriorityQueue {
  constructor() {
    this.heap = [];
    this.count = 0;
  }

  get length() {
    return this.count;
  }

  push(item) {
    this.count += 1;
    this.heap.push(item);
    this._shiftUp(this.count - 1);
  }

  pop() {
    const item = this.heap[0];
    this.count -= 1;
    if (this.count === 0) {
      this.heap = [];
    } else {
      this.heap[0] = this.heap.pop();
      this._shiftDown(0);
    }
    return item;
  }

  top() {
    return this.heap[0];
  }

  _shiftUp(index) {
    const parent = Math.floor((index - 1) / 2);
    if (index > 0 && this.heap[index] > this.heap[parent]) {
      [this.heap[index], this.heap[parent]] = [this.heap[parent], this.heap[index]];
      this._shiftUp(parent);
    }
  }

  _shiftDown(index) {
    const leftChild = index * 2 + 1;
    const rightChild = index * 2 + 2;
    let maxChild = index;
    if (leftChild < this.count && this.heap[leftChild] > this.heap[maxChild]) {
      maxChild = leftChild;
    }
    if (rightChild < this.count && this.heap[rightChild] > this.heap[maxChild]) {
      maxChild = rightChild;
    }
    if (maxChild !== index) {
      [this.heap[index], this.heap[maxChild]] = [this.heap[maxChild], this.heap[index]];
      this._shiftDown(maxChild);
    }
  }
}
