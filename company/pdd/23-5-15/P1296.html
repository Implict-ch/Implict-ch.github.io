# 题目描述

塔子哥接到了一个给音乐会门票编号的任务。每次可选的门票编号区间是[L,R][L,R][L,R]。

由于音乐会的日期定在555月202020号，塔子哥认为包含555，202020和520520520的编号是有特殊含义，准备保留给对应的门票。例如：123451234512345，其中包含了555，是一个特殊编号；而246782467824678就是一个普通编号。

现在塔子哥想知道，在可选的门票编号区间内，有多少符合上面要求的特殊编号。

# 输入描述

输入第一行一个整数 TTT，表示可选的门票编码区间个数。( 1≤T≤10001 \leq T \leq 10001≤T≤1000 )

接下来T行，每行2个整数: LLL 和 RRR ，表示编码可选的区间(闭区间，即包括 LLL 和 RRR 。( 1≤L≤R≤1061 \leq L \leq R \leq 10^61≤L≤R≤10​6​​ )

# 输出描述

输出一共TTT行，每行333个整数，分别表示对应区间里的5、20和520520520的编号数量

# 样例

样例输入

3
5 20
10 100
52 6000

样例输出

2 1 0
18 1 0
2348 218 16

样例说明

对于第一组可选编号区间:[5,20][5,20][5,20]，符合要求的特殊编号包括:5,15,20{ 5,15,20}5,15,20，其中包括555的有222个，202020的111个，520520520的000个。

对于第三组可选编号区间:[52,1000]，包含520的特殊编号包括:{ 520，1520，2520，3520，4520，5200 , 5201 , 5202 , 5203, 5204, 5205 , 5206 , 5207 , 5208 , 5209 , 5520}，共161616个。

# 思路：前缀和+预处理

对于一次查询 L,RL, RL,R，因为数据范围为 1≤L,R≤1061\leq L,R\leq 10^61≤L,R≤10​6​​。 对于 [L,R][L,R][L,R] 内的每个数，查询其是否存在 555 的次数为 R−L+1R-L+1R−L+1，同时还取决于这个数的长度。 长度为 111 的数有 999 个，长度为 222 的数有 909090 个，长度为 333 的数有 900900900 个，长度为 444 的数有 900090009000 个，长度为 555 的数有 900009000090000 个，长度为 666 的数 900000900000900000 个，长度为 777 的数至多就 100000010000001000000 这一个数。

假设每个数都得完全遍历，总的遍历次数为：7+∑i=169×10i−1=5888890≈6×1067+\sum\limits_{i=1}^6 9\times 10^{i-1} = 5888890\approx6\times 10^67+​i=1​∑​6​​9×10​i−1​​=5888890≈6×10​6​​。

而 TTT 组数据，最多有 100010001000 组，故需要考虑优化。

定义 cnt[i][0]cnt[i][0]cnt[i][0] 表示 [1,i][1,i][1,i] 中存在 555 的数的个数。 则求解 [L,R][L,R][L,R] 中 555 的数的个数即为：cnt[R][0]−cnt[L−1][0]cnt[R][0] - cnt[L-1][0]cnt[R][0]−cnt[L−1][0]，对于 202020 和 520520520 同理。

这样，一次预处理后，每组数据只需要 O(1)O(1)O(1) 时间即可。

# CPP

#include <bits/stdc++.h>
using namespace std;

const int MAXV = 1000001;
// cnt[i][0] 表示前i个数中有5的数的个数
// cnt[i][1] 表示前i个数中有20的数的个数
// cnt[i][2] 表示前i个数中有520的数的个数
int cnt[MAXV][3];

void solve() {

    int l, r;
    cin >> l >> r;

    // 直接前缀和相减即区间和
    cout << (cnt[r][0] - cnt[l - 1][0]) << " " << (cnt[r][1] - cnt[l - 1][1]) << " " << (cnt[r][2] - cnt[l - 1][2]) << "\n";
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    // 预处理前i个数中有5、20、520的数的个数
    for (int i = 1; i < MAXV; ++i) {
        string s = to_string(i);
        int c[3] = {0};
        for (int j = 0; j < s.size(); ++j) {
            // 注意这里每个数只统计一次，所有出现过标记为 1 即可
            if (s[j] == '5') c[0] = 1;
            if (j + 1 < s.size() && s[j] == '2' && s[j + 1] == '0') c[1] = 1;
            if (j + 2 < s.size() && s[j] == '5' && s[j + 1] == '2' && s[j + 2] == '0') c[2] = 1;
        }

        // 累加当前数是否出现过5、20、520即可
        for (int j = 0; j < 3; ++j) cnt[i][j] = cnt[i - 1][j] + c[j];
    }

    int T = 1;
    cin >> T;
    while (T--) solve();

    return 0;
}

import java.util.Scanner;

public class Main {
    static final int MAXV = 1000001;
    // cnt[i][0] 表示前i个数中有5的数的个数
    // cnt[i][1] 表示前i个数中有20的数的个数
    // cnt[i][2] 表示前i个数中有520的数的个数
    static int[][] cnt = new int[MAXV][3];

    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);

        // 预处理前i个数中有5、20、520的数的个数
        for (int i = 1; i < MAXV; ++i) {
            String s = Integer.toString(i);
            int[] c = new int[3];
            for (int j = 0; j < s.length(); ++j) {
                // 注意这里每个数只统计一次，所有出现过标记为 1 即可
                if (s.charAt(j) == '5') c[0] = 1;
                if (j + 1 < s.length() && s.charAt(j) == '2' && s.charAt(j + 1) == '0') c[1] = 1;
                if (j + 2 < s.length() && s.charAt(j) == '5' && s.charAt(j + 1) == '2' && s.charAt(j + 2) == '0') c[2] = 1;
            }

            // 累加当前数是否出现过5、20、520即可
            for (int j = 0; j < 3; ++j) cnt[i][j] = cnt[i - 1][j] + c[j];
        }

        int T = scanner.nextInt();
        while (T-- > 0) solve(scanner);
    }

    public static void solve(Scanner scanner) {

        int l = scanner.nextInt();
        int r = scanner.nextInt();

        // 直接前缀和相减即区间和
        System.out.println((cnt[r][0] - cnt[l - 1][0]) + " " + (cnt[r][1] - cnt[l - 1][1]) + " " + (cnt[r][2] - cnt[l - 1][2]));
    }
}
