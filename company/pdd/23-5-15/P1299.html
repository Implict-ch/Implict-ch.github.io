# 题目描述

塔子哥最近接到一个游戏设计的工程，其中有 NNN 个角色（分别编号 1→N1\rightarrow N1→N ）。

在游戏设定中，每个初始角色的战斗力用其等级高低来评价，每个初始角色都会有一个固定的等级。

为了设计一个完善的等级系统，塔子哥为这些初始角色设计了 MMM 场战斗来测试这些初始角色的强度。如果战斗的角色等级相等，那么会出现平局；如果不同等级角色之间的战斗，高等级的角色一定会击败低等级角色。

塔子哥想知道应该为这些角色设计多少种等级数量 ansansans ，如果战斗违反设定或者无法确定，那么就返回 ”−1-1−1“。

# 输入描述

输入第一行一个整数 TTT ，代表测试用例的组数。(1≤T≤101 \leq T \leq 101≤T≤10)

接下来 TTT 组测试数据，对于每一组测试用例，第一行两个整数 NNN 和 MMM ，分别代表角色的数量和战斗的场数。( 1≤N≤10000,1≤M≤200001\leq N \leq 10000,1 \leq M \leq 200001≤N≤10000,1≤M≤20000) (可能存在重边)

接下来 MMM 行，表示角色A和角色B的战斗结果( 1≤A,B≤N1 \leq A,B \leq N1≤A,B≤N )，有三种可能:

  1. A>BA>BA>B，表示AAA击败了BBB;

  2. A<BA<BA<B，表示AAA输给了BBB;

  3. A=BA=BA=B，表示AAA和BBB打平;

# 输出描述

对于每组测试用例，如果战斗结果符合等级设定，输出总共的等级数量 ansansans ; 否则输出−1-1−1，表示违反设定或无法确定。

样例说明

第二组用例，222击败了111，222又击败了333,但是无法确定111和333之间的等级关系，因此无法确定人物之间的等级。

第三组用例，111号自己输给自己，违反了相同等级不能分出胜负的设定。

第四组用例，有:{111}<{222}<{333}<{444}>{555}<{6,76,76,7}，共666种等级。

第五组用例，333没有任何战斗，无法确定其所在等级。

# 思路：拓扑排序+并查集

首先，考虑给定的战斗结果是合理的情况下：

  • 每个打平的角色之间的战斗值必然是相等的，故战斗结果打平的这些角色可以认为是一个集合。每个不同集合中的角色的战斗可以看成是集合之间的战斗，将这个战斗结果抽象成一个图，一次战斗中，战斗结果为失败的集合会指向战斗结果为成功的集合。如此，必然存在唯一一个集合 startstartstart，图中没有指向 startstartstart 的集合。这样就是一个拓扑排序的模板题了。

接下来考虑可能的不合理的情况：

  • 存在两个及以上的集合，在图中都没有指向他们的其他集合，即图中有两个点的入度为000。此时在拓扑排序结束后，必然有集合未被访问到，所以直接统计访问到的集合与集合总数是否相等即可。

# CPP

#include <bits/stdc++.h>
using namespace std;

typedef pair<int, int> PII;

// 并查集类
struct UnionSet {
    int n;
    vector<int> p;

    UnionSet(int n): n(n), p(n) {
        iota(p.begin(), p.end(), 0);
    }

    int find(int x) {
        return x == p[x] ? p[x] : (p[x] = find(p[x]));
    }

    bool merge(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return false;
        p[py] = px;
        return true;
    }
};

void solve() {
    int n, m;
    cin >> n >> m;

    int a, b;
    string op;

    vector<PII> edges;

    // 并查集
    UnionSet us(n);

    // 合并每个战斗结果相等的点，构成集合
    for (int i = 0; i < m; ++i) {
        cin >> a >> op >> b;
        a--, b--;
        if (op == "=") {
            us.merge(a, b);
        } else {
            if (op == ">") {
                swap(a, b);
            }
            // 存储每条边，此时集合暂未构建完毕，先存储，等待集合构建完毕再建图
            edges.emplace_back(a, b);
        }
    }

    vector<int> d(n);
    vector<vector<int>> g(n);

    for (auto& [u, v]: edges) {
        // 点 u 所在的集合指向点 v 所在的集合
        g[us.find(u)].emplace_back(us.find(v));
        // 点 v 所在的集合入度 + 1
        d[us.find(v)] += 1;
    }

    vector<int> level(n);
    queue<int> q;

    // 统计集合数量，如果有两个及以上的集合入度为 0，说明这两个集合的战斗结果不确定，所以必然无法确定答案
    int cnt_set = 0;
    for (int i = 0; i < n; ++i)
        if (i == us.find(i)) {
            level[i] = 1;
            cnt_set += 1;
            if (d[i] == 0) {
                if (!q.empty()) {
                    cout << "-1\n";
                    return;
                }
                q.push(i);
            }
        }

    // 拓扑排序模板
    int cnt_q = 0;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        // 统计访问到的集合的数量
        cnt_q += 1;

        for (int v: g[u]) {
            level[v] = max(level[v], level[u] + 1);
            if (--d[v] == 0) {
                q.push(v);
            }
        }
    }

    // 如果未访问完所有的集合，则返回 -1
    if (cnt_q < cnt_set) {
        cout << "-1\n";
    } else {
        // 有多少个集合，就有多少个等级
        cout << cnt_set << "\n";
    }
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T = 1;
    cin >> T;
    while (T--) solve();

    return 0;
}

# 并查集类
class UnionSet:
    def __init__(self, n: int):
        self.n = n
        self.p = list(range(n))

    def find(self, x: int) -> int:
        if x == self.p[x]:
            return x
        self.p[x] = self.find(self.p[x])
        return self.p[x]

    def merge(self, x: int, y: int) -> bool:
        px, py = self.find(x), self.find(y)
        if px == py:
            return False
        self.p[py] = px
        return True

def solve() -> None:
    n, m = map(int, input().split())

    edges = []
    us = UnionSet(n)

    # 合并每个战斗结果相等的点，构成集合
    for i in range(m):
        a, op, b = input().split()
        a, b = int(a) - 1, int(b) - 1
        if op == "=":
            us.merge(a, b)
        else:
            if op == ">":
                a, b = b, a
            # 存储每条边，此时集合暂未构建完毕，先存储，等待集合构建完毕再建图
            edges.append((a, b))

    d = [0] * n
    g = [[] for _ in range(n)]

    for u, v in edges:
        # 点 u 所在的集合指向点 v 所在的集合
        g[us.find(u)].append(us.find(v))
        # 点 v 所在的集合入度 + 1
        d[us.find(v)] += 1

    level = [0] * n
    q = []

    # 统计集合数量，如果有两个及以上的集合入度为 0，说明这两个集合的战斗结果不确定，所以必然无法确定答案
    cnt_set = 0
    for i in range(n):
        if i == us.find(i):
            level[i] = 1
            cnt_set += 1
            if d[i] == 0:
                if q:
                    print("-1")
                    return
                q.append(i)

    # 拓扑排序模板
    cnt_q = 0
    while q:
        u = q.pop(0)
        # 统计访问到的集合的数量
        cnt_q += 1

        for v in g[u]:
            level[v] = max(level[v], level[u] + 1)
            d[v] -= 1
            if d[v] == 0:
                q.append(v)

    # 如果未访问完所有的集合，则返回 -1
    if cnt_q < cnt_set:
        print("-1")
    else:
        # 有多少个集合，就有多少个等级
        print(cnt_set)

if __name__ == '__main__':
    T = int(input())
    for _ in range(T):
        solve()

        for (int i = 0; i < m; i++) {
            a = sc.nextInt() - 1;
            op = sc.next();
            b = sc.nextInt() - 1;
            if (op.equals("=")) {

                    a = b;

                edges.add(new PII(a, b));

            int u = us.find(e.first);
            int v = us.find(e.second);
            // 点 u 所在的集合指向点 v 所在的集合
            g[u].add(v);
            // 点 v 所在的集合入度 + 1
            d[v]++;
        }
        int[] level = new int[n];
        Queue<Integer> q = new LinkedList<>();
        // 统计集合数量，如果有两个及以上的集合入度为 0，说明这两个集合的战斗结果不确定，所以必然无法确定答案
        int cnt_set = 0;

                    q.offer(v);
                }

# Go

package main

import (
    "fmt"
)

type PII struct {
    first, second int
}

// 并查集类
type UnionSet struct {
    n int
    p []int
}

func NewUnionSet(n int) *UnionSet {
    p := make([]int, n)
    for i := 0; i < n; i++ {
        p[i] = i
    }
    return &UnionSet{n, p}
}

func (us *UnionSet) find(x int) int {
    if x == us.p[x] {
        return x
    }
    us.p[x] = us.find(us.p[x])
    return us.p[x]
}

func (us *UnionSet) merge(x, y int) bool {
    px, py := us.find(x), us.find(y)
    if px == py {
        return false
    }
    us.p[py] = px
    return true
}

func solve() {
    var n, m int
    fmt.Scan(&n, &m)

    edges := make([]PII, 0)
    us := NewUnionSet(n)

    // 合并每个战斗结果相等的点，构成集合
    for i := 0; i < m; i++ {
        var a, b int
        var op string
        fmt.Scan(&a, &op, &b)
        a, b = a-1, b-1
        if op == "=" {
            us.merge(a, b)
        } else {
            if op == ">" {
                a, b = b, a
            }
            // 存储每条边，此时集合暂未构建完毕，先存储，等待集合构建完毕再建图
            edges = append(edges, PII{a, b})
        }
    }

    d := make([]int, n)
    g := make([][]int, n)

    for _, e := range edges {
        u, v := e.first, e.second
        // 点 u 所在的集合指向点 v 所在的集合
        g[us.find(u)] = append(g[us.find(u)], us.find(v))
        // 点 v 所在的集合入度 + 1
        d[us.find(v)] += 1
    }

    level := make([]int, n)
    q := make([]int, 0)

    // 统计集合数量，如果有两个及以上的集合入度为 0，说明这两个集合的战斗结果不确定，所以必然无法确定答案
    cntSet := 0
    for i := 0; i < n; i++ {
        if i == us.find(i) {
            level[i] = 1
            cntSet += 1
            if d[i] == 0 {
                if len(q) != 0 {
                    fmt.Println("-1")
                    return
                }
                q = append(q, i)
            }
        }
    }

    // 拓扑排序模板
    cntQ := 0
    for len(q) != 0 {
        u := q[0]
        q = q[1:]
        // 统计访问到的集合的数量
        cntQ += 1

        for _, v := range g[u] {
            level[v] = max(level[v], level[u]+1)
            d[v] -= 1
            if d[v] == 0 {
                q = append(q, v)
            }
        }
    }

    // 如果未访问完所有的集合，则返回 -1
    if cntQ < cntSet {
        fmt.Println("-1")
    } else {
        // 有多少个集合，就有多少个等级
        fmt.Println(cntSet)
    }
}

func main() {
    var T int
    fmt.Scan(&T)
    for i := 0; i < T; i++ {
        solve()
    }
}

    if a > b {
        return a
    }
    return b
}

    const lines = input.trim().split('\n');
    let index = 0;

    function readLine() {
        return lines[index++];
    }

    class UnionSet {
        constructor(n) {
            this.n = n;
            this.p = new Array(n).fill().map((_, i) => i);
        }

        find(x) {
            if (x === this.p[x]) {
                return x;
            }
            this.p[x] = this.find(this.p[x]);
            return this.p[x];
        }

        merge(x, y) {
            const px = this.find(x), py = this.find(y);
            if (px === py) {
                return false;
            }
            this.p[py] = px;
            return true;
        }
    }

    function solve() {
        const [n, m] = readLine().split(' ').map(Number);

        const edges = [];
        const us = new UnionSet(n);

        // 合并每个战斗结果相等的点，构成集合
        for (let i = 0; i < m; i++) {

                us.merge(x, y);

                        return;

                    q.push(i);
                }

                level[v] = Math.max(level[v], level[u] + 1);

                if (d[v] === 0) {

                }

        } else {
            // 有多少个集合，就有多少个等级
