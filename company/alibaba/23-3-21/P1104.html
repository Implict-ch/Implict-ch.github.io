塔子哥有两个机器，一个是他自己的电脑，另一个是他队友黑白胖球之一CerqCerqCerq的电脑。

const int mod = 1000000007;
vector<int> fac(200006);
// 快速幂 求解 逆元
long long ksm(long long a, int b){
    long long ans = 1;
    long long base = a % mod;
    while(b != 0){
        if(b & 1){
            ans = ans * base % mod;
        }
        base = base * base % mod;
        b >>= 1;
    }
    return ans;
}
// 求组合数C(n , m)
int comb(int n, int m){
    if(m > n){
        return 0;
    }
    long long x = fac[n];
    long long y = fac[m] * fac[n - m] % mod;
    return x * ksm(y, mod - 2) % mod;
}

int getMethods(int cost, int k){
    // 预处理阶乘
    fac[0] = 1;
    for(int i = 1; i <= 200005; i++){
        fac[i] = fac[i-1] * i % mod;

    for(int i = 1; i < k; i++){
        ans = (ans + comb(n - 1, i - 1) * comb(n - 1, k - i - 1) % mod) % mod;
    }
    return ans;
}

# Js

特别注意,与其他语言不同的是，js的最大整数为253−1≈9e152^{53}-1 \approx 9e152​53​​−1≈9e15 ,那就意味着1e9+71e9+71e9+7 这个模数下即使是正确的取模方式，相乘操作也会导致溢出。所以所有乘法需要替换成快速乘 来将乘法转加法，这样就一定不会溢出。

由于O(1)O(1)O(1)的乘法我们使用快速乘代替(实现类似于快速幂)，那么我们的总复杂度也需要乘上一个logmod≈31log\ mod \approx 31logmod≈31 的常数。

function getMethods(cost, k) {
  const mod = 1000000007;
  var fac = [1];
  
  // 封装快速乘，实现类似快速幂
  function mult(a, b) {
    let ans = 0;
    let base = a;
    while (b != 0) {
      if ((b & 1) == 1) {
        ans = (ans + base) % mod;
      }
      base = (base + base) % mod;
      b >>= 1;
    }
    return ans;
  }

  for (let i = 1; i < 200006; i++) {
    fac.push(mult(fac[i-1], i));
  }
  // 快速幂
  function ksm(a, b) {
    let ans = 1;
    let base = a;
    while (b != 0) {
      if ((b & 1) == 1) {
        ans = mult(ans, base);
      }
      base = mult(base, base);
      b >>= 1;
    }
    return ans;
  }

  if (cost % 2 !== 0) {
    return 0;
  }
  // 求组合数
  function comb(n, m) {
    if (m > n) {
      return 0;
    }
    const x = fac[n];
    const y = mult(fac[m], fac[n-m]);
    return mult(x, ksm(y, mod-2));
  }

  const n = cost / 2;
  let ans = 0;
  // 根据题解公式，求解答案
  for (let i = 1; i < k; i++) {
    ans = (ans + mult(comb(n-1, i-1), comb(n-1, k-i-1))) % mod;
  }
  return ans;
}
