# 思路：模拟

根据题目的三个条件进行模拟即可。

  • 中位数不低于 a
  • 平均考核成绩不低于 b
  • 去掉最高分和最低分后的平均考核成绩不低于 c

这里有一个 trick ，如果求平均值，会产生小数，那么就会有精度问题。

由于这里的值域数据范围都不大，所以可以将除法修改为乘法。

比如：xy≥n\frac{x}{y}\geq n​y​​x​​≥n 等价于判断 x≥y×nx \geq y\times nx≥y×n

时间复杂度：O(n)O(n)O(n)

# C++

#include <bits/stdc++.h>
using namespace std;

void solve() {
    int n;
    cin >> n;
    vector<int> s(n);
    for (int i = 0; i < n; ++i) cin >> s[i];
    
    int a, b, c;
    cin >> a >> b >> c;

    sort(s.begin(), s.end());

    bool ok = false;
    // 计算中位数，如果是奇数，则是最中间的，如果是偶数，则是中间两个的平均数
    if (n % 2 == 1) {
        if (s[n / 2] >= a) {
            ok = true;
        }
    } else {
        // (x + y) / 2 >= a ====> (x + y) >= 2 * a
        if (s[n / 2 - 1] + s[n / 2] >= 2 * a) {
            ok = true;
        }
    }

    // 计算平均数，(s1+...+sn) >= b*n
    if (accumulate(s.begin(), s.end(), 0) >= b * n) {
        ok = true;
    }

    // 计算去掉最高和最低的平均数 (s1+...+sn-max-min) >= c * (n-2)
    if (accumulate(s.begin(), s.end(), 0) - s[0] - s[n - 1] >= c * (n - 2)) {
        ok = true;
    }

    puts(ok ? "Yes" : "No");
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;

    for (int i = 1; i <= T; ++i) {
        solve();
    }

    return 0;
}

# Go

package main

import (
    "fmt"
    "sort"
)

func solve() {
    var n, a, b, c int
    fmt.Scan(&n)
    s := make([]int, n)
    for i := 0; i < n; i++ {
        fmt.Scan(&s[i])
    }

    fmt.Scan(&a, &b, &c)

    sort.Ints(s)

    ok := false
    // 计算中位数，如果是奇数，则是最中间的，如果是偶数，则是中间两个的平均数
    if n % 2 == 1 {
        if s[n / 2] >= a {
            ok = true
        }
    } else {
        // (x + y) / 2 >= a ====> (x + y) >= 2 * a
        if s[n / 2 - 1] + s[n / 2] >= 2 * a {
            ok = true
        }
    }

    sum := 0
    for i := 0; i < n; i++ {
        sum += s[i]
    }
    // 计算平均数，(s1+...+sn) >= b*n
    if sum >= b * n {
        ok = true
    }

    // 计算去掉最高和最低的平均数 (s1+...+sn-max-min) >= c * (n-2)
    if sum-s[0]-s[n-1] >= c * (n - 2) {
        ok = true
    }

    if ok {
        fmt.Println("Yes")
    } else {
        fmt.Println("No")
    }
}

func main() {
    var T int
    fmt.Scanf("%d", &T)

    for i := 1; i <= T; i++ {
        solve()
    }
}

process.stdin.resume();
process.stdin.setEncoding('utf-8');
let input = '';
process.stdin.on('data', (data) => {
    input += data;
    return;

process.stdin.on('end', () => {
    const lines = input.trim().split('\n');
    let index = 0;
    const T = parseInt(lines[index++]);

    for (let i = 0; i < T; i++) {
        const n = parseInt(lines[index++]);
        const s = lines[index++].split(' ').map(Number);
        const [a, b, c] = lines[index++].split(' ').map(Number);

        s.sort((a, b) => a - b);

        let ok = false;
        // 计算中位数，如果是奇数，则是最中间的，如果是偶数，则是中间两个的平均数
        if (n % 2 === 1) {
            if (s[Math.floor(n / 2)] >= a) {
                ok = true;
            }
        } else {
            // (x + y) / 2 >= a ====> (x + y) >= 2 * a
            if (s[Math.floor(n / 2) - 1] + s[Math.floor(n / 2)] >= 2 * a) {
                ok = true;
            }
        }

        // 计算平均数，(s1+...+sn) >= b*n
        if (s.reduce((acc, cur) => acc + cur, 0) >= b * n) {
            ok = true;
        }

        // 计算去掉最高和最低的平均数 (s1+...+sn-max-min) >= c * (n-2)
        if (s.slice(1, n - 1).reduce((acc, cur) => acc + cur, 0) >= c * (n - 2)) {
            ok = true;
        }

        console.log(ok ? "Yes" : "No");
    }
});
